237static inline void file2Json(std::string filename, Json::Value& js, enum comments comm = file::comments::are_discarded, enum error err = file::error::is_throw)
238{
239 Json::CharReaderBuilder parser;
240 if( comments::are_forbidden == comm )
241 Json::CharReaderBuilder::strictMode( &parser.settings_);
242 else if( comments::are_discarded == comm )
243 {
244 Json::CharReaderBuilder::strictMode( &parser.settings_);
245 // workaround for a linker bug in jsoncpp from package manager
246 Json::Value js_true (true);
247 Json::Value js_false (false);
248 parser.settings_["allowComments"].swap( js_true);
249 parser.settings_["collectComments"].swap(js_false);
250 }
251 else
252 Json::CharReaderBuilder::setDefaults( &parser.settings_);
253
254 std::ifstream isI( filename);
255 if( !isI.good())
256 {
257 std::string message = "\nAn error occured while parsing "+filename+"\n";
258 message += "*** File does not exist! *** \n\n";
259 if( err == error::is_throw)
260 throw std::runtime_error( message);
261 else if (err == error::is_warning)
262 std::cerr << "WARNING: "<<message<<std::endl;
263 else
264 ;
265 return;
266 }
267 std::string errs;
268 if( !parseFromStream( parser, isI, &js, &errs) )
269 {
270 std::string message = "An error occured while parsing "+filename+"\n"+errs;
271 if( err == error::is_throw)
272 throw std::runtime_error( message);
273 else if (err == error::is_warning)
274 std::cerr << "WARNING: "<<message<<std::endl;
275 else
