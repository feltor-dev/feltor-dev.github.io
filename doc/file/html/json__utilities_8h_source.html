310static inline void file2Json(std::string filename, Json::Value& js, enum comments comm = file::comments::are_discarded, enum error err = file::error::is_throw)
311{
312 Json::CharReaderBuilder parser;
313 if( comments::are_forbidden == comm )
314 Json::CharReaderBuilder::strictMode( &parser.settings_);
315 else if( comments::are_discarded == comm )
316 {
317 Json::CharReaderBuilder::strictMode( &parser.settings_);
318 // workaround for a linker bug in jsoncpp from package manager
319 Json::Value js_true (true);
320 Json::Value js_false (false);
321 parser.settings_["allowComments"].swap( js_true);
322 parser.settings_["collectComments"].swap(js_false);
323 }
324 else
325 Json::CharReaderBuilder::setDefaults( &parser.settings_);
326
327 std::ifstream isI( filename);
328 if( !isI.good())
329 {
330 std::string message = "\nAn error occured while parsing "+filename+"\n";
331 message += "*** File does not exist! *** \n\n";
332 if( err == error::is_throw)
333 throw std::runtime_error( message);
334 else if (err == error::is_warning)
335 std::cerr << "WARNING: "<<message<<std::endl;
336 else
337 ;
338 return;
339 }
340 std::string errs;
341 if( !parseFromStream( parser, isI, &js, &errs) )
342 {
343 std::string message = "An error occured while parsing "+filename+"\n"+errs;
344 if( err == error::is_throw)
345 throw std::runtime_error( message);
346 else if (err == error::is_warning)
347 std::cerr << "WARNING: "<<message<<std::endl;
348 else
