73struct DSField
74{
75 DSField() = default;
76 //z component of v may not vanish
77 DSField( const dg::geo::CylindricalVectorLvl1& v,
78 const dg::aGeometry2d& g ):
79 m_g(g)
80 {
81 dg::HVec v_zeta, v_eta;
82 dg::pushForwardPerp( v.x(), v.y(), v_zeta, v_eta, g);
83 dg::HVec vx = dg::pullback( v.x(), g);
84 dg::HVec vy = dg::pullback( v.y(), g);
85 dg::HVec vz = dg::pullback( v.z(), g);
86 dg::HVec divvvz = dg::pullback( v.divvvz(), g);
87 dg::blas1::pointwiseDivide(v_zeta, vz, v_zeta);
88 dg::blas1::pointwiseDivide(v_eta, vz, v_eta);
89 dzetadphi_ = dg::forward_transform( v_zeta, g );
90 detadphi_ = dg::forward_transform( v_eta, g );
91 dvdphi_ = dg::forward_transform( divvvz, g );
92 }
93 //interpolate the vectors given in the constructor on the given point
94 void operator()(double t, const std::array<double,3>& y, std::array<double,3>& yp) const
95 {
96 // shift point into domain
97 yp[0] = interpolate(dg::lspace, dzetadphi_, y[0], y[1], *m_g);
98 yp[1] = interpolate(dg::lspace, detadphi_, y[0], y[1], *m_g);
99 yp[2] = interpolate(dg::lspace, dvdphi_, y[0], y[1], *m_g)*y[2];
100 }

103 dg::ClonePtr<dg::aGeometry2d> m_g;

112template<class real_type>
113void integrate_all_fieldlines2d( const dg::geo::CylindricalVectorLvl1& vec,
114 const dg::aRealGeometry2d<real_type>& grid_field,
115 const dg::aRealTopology2d<real_type>& grid_evaluate,
116 std::array<thrust::host_vector<real_type>,3>& yp,
117 const thrust::host_vector<double>& vol0,
118 thrust::host_vector<real_type>& yp2b,
119 thrust::host_vector<bool>& in_boxp,
120 real_type deltaPhi, real_type eps)
121{
122 //grid_field contains the global geometry for the field and the boundaries
123 //grid_evaluate contains the points to actually integrate
124 std::array<thrust::host_vector<real_type>,3> y{
125 dg::evaluate( dg::cooX2d, grid_evaluate),
126 dg::evaluate( dg::cooY2d, grid_evaluate),
127 vol0
128 };
129 yp.fill(dg::evaluate( dg::zero, grid_evaluate));
130 //construct field on high polynomial grid, then integrate it
131 dg::geo::detail::DSField field;
132 if( !dynamic_cast<const dg::CartesianGrid2d*>( &grid_field))
133 field = dg::geo::detail::DSField( vec, grid_field);
134
135 //field in case of cartesian grid
136 dg::geo::detail::DSFieldCylindrical4 cyl_field(vec);
137 const unsigned size = grid_evaluate.size();
138 dg::Adaptive<dg::ERKStep<std::array<real_type,3>>> adapt(
139 "Dormand-Prince-7-4-5", std::array<real_type,3>{0,0,0});
140 dg::AdaptiveTimeloop< std::array<real_type,3>> odeint;
141 if( dynamic_cast<const dg::CartesianGrid2d*>( &grid_field))
142 odeint = dg::AdaptiveTimeloop<std::array<real_type,3>>( adapt,
143 cyl_field, dg::pid_control, dg::fast_l2norm, eps, 1e-10);
144 else
145 odeint = dg::AdaptiveTimeloop<std::array<real_type,3>>( adapt,
146 field, dg::pid_control, dg::fast_l2norm, eps, 1e-10);
147
148 for( unsigned i=0; i<size; i++)
149 {
150 std::array<real_type,3> coords{y[0][i],y[1][i],y[2][i]}, coordsP;
151 //x,y,s
152 real_type phi1 = deltaPhi;
153 odeint.set_dt( deltaPhi/2.);
154 odeint.integrate( 0, coords, phi1, coordsP);
155 yp[0][i] = coordsP[0], yp[1][i] = coordsP[1], yp[2][i] = coordsP[2];
156 }
157 yp2b.assign( grid_evaluate.size(), deltaPhi); //allocate memory for output
158 in_boxp.resize( yp2b.size());
159 //Now integrate again but this time find the boundary distance
160 for( unsigned i=0; i<size; i++)
161 {
162 std::array<real_type,3> coords{y[0][i],y[1][i],y[2][i]}, coordsP;
163 in_boxp[i] = grid_field.contains( yp[0][i], yp[1][i]) ? true : false;
164 if( false == in_boxp[i])
165 {
166 //x,y,s
167 real_type phi1 = deltaPhi;
168 odeint.integrate_in_domain( 0., coords, phi1, coordsP, 0., (const
169 dg::aRealTopology2d<real_type>&)grid_field, eps);
170 yp2b[i] = phi1;
171 }
172 }
173}

268struct WallFieldlineCoordinate : public aCylindricalFunctor<WallFieldlineCoordinate>
269{
271 WallFieldlineCoordinate(
272 const dg::geo::CylindricalVectorLvl0& vec,
273 const dg::aRealTopology2d<double>& domain,
274 double maxPhi, double eps, std::string type) :
275 m_domain( domain), m_cyl_field(vec),
276 m_deltaPhi( maxPhi), m_eps( eps), m_type(type)
277 {

279 throw std::runtime_error( "Distance type "+m_type+" not recognized!\n");
280 }
281 double do_compute( double R, double Z) const
282 {
283 double phiP = m_deltaPhi, phiM = -m_deltaPhi;
284 std::array<double,3> coords{ R, Z, 0}, coordsP(coords), coordsM(coords);
285 // determine sign
286 m_cyl_field( 0., coords, coordsP);
287 double sign = coordsP[2] > 0 ? +1. : -1.;
288 try{
289 dg::AdaptiveTimeloop<std::array<double,3>> odeint(
290 dg::Adaptive<dg::ERKStep<std::array<double,3>>>(
291 "Dormand-Prince-7-4-5", coords), m_cyl_field,
292 dg::pid_control, dg::fast_l2norm, m_eps, 1e-10);
293 odeint.integrate( 0., coords, phiP, coordsP);
294 odeint.integrate( 0., coords, phiM, coordsM);
295 }catch (std::exception& e)
296 {
297 // if not possible the distance is large
298 phiP = m_deltaPhi;
299 coordsP[2] = 1e6*phiP;
300 phiM = -m_deltaPhi;
301 coordsM[2] = 1e6*phiM;
302 }
303 if( m_type == "phi")
304 return sign*(-phiP-phiM)/(phiP-phiM);
305 double sP = coordsP[2], sM = coordsM[2];
306 double value = sign*(-sP-sM)/(sP-sM);
307 if( (phiM <= -m_deltaPhi) && (phiP >= m_deltaPhi))
308 return 0.; //return exactly zero if sheath not reached
309 if( (phiM <= -m_deltaPhi))
310 return value*sign > 0 ? value : 0.; // avoid false negatives
311 if( (phiP >= m_deltaPhi))
312 return value*sign < 0 ? value : 0.; // avoid false positives
313 return value;
314 }

318 dg::geo::detail::DSFieldCylindrical3 m_cyl_field;

792template<class G, class I, class container>
793container Fieldaligned<G, I,container>::interpolate_from_coarse_grid(
794 const G& grid, const container& in)
795{
796 //I think we need grid as input to split input vector and we need to interpret
797 //the grid nodes as node centered not cell-centered!
798 //idea: apply I+/I- cphi - 1 times in each direction and then apply interpolation formula
799 assert( m_g->Nz() % grid.Nz() == 0);
800 unsigned Nz_coarse = grid.Nz(), Nz = m_g->Nz();
801 unsigned cphi = Nz / Nz_coarse;
802
803 container out = dg::evaluate( dg::zero, *m_g);
804 container helper = dg::evaluate( dg::zero, *m_g);
805 dg::split( helper, m_temp, *m_g);
806 std::vector<dg::View< container>> out_split = dg::split( out, *m_g);
807 std::vector<dg::View< const container>> in_split = dg::split( in, grid);
808 for ( int i=0; i<(int)Nz_coarse; i++)
809 {
810 //1. copy input vector to appropriate place in output
811 dg::blas1::copy( in_split[i], out_split[i*cphi]);
812 dg::blas1::copy( in_split[i], m_temp[i*cphi]);
813 }
814 //Step 1 needs to finish so that m_temp contains values everywhere
815 //2. Now apply plus and minus T to fill in the rest
816 for ( int i=0; i<(int)Nz_coarse; i++)
817 {
818 for( int j=1; j<(int)cphi; j++)
819 {
821 dg::blas2::symv( m_minus, out_split[i*cphi+j-1], out_split[i*cphi+j]);
823 dg::blas2::symv( m_plus, m_temp[(i*cphi+cphi+1-j)%Nz], m_temp[i*cphi+cphi-j]);
824 }
825 }
826 //3. Now add up with appropriate weights
827 for( int i=0; i<(int)Nz_coarse; i++)
828 for( int j=1; j<(int)cphi; j++)
829 {
830 double alpha = (double)(cphi-j)/(double)cphi;
831 double beta = (double)j/(double)cphi;
832 dg::blas1::axpby( alpha, out_split[i*cphi+j], beta, m_temp[i*cphi+j], out_split[i*cphi+j]);
833 }
834 return out;
835}
836template<class G, class I, class container>
837void Fieldaligned<G, I,container>::integrate_between_coarse_grid( const G& grid, const container& in, container& out)
838{
839 assert( m_g->Nz() % grid.Nz() == 0);
840 unsigned Nz_coarse = grid.Nz(), Nz = m_g->Nz();
841 unsigned cphi = Nz / Nz_coarse;
842
843 out = in;
844 container helperP( in), helperM(in), tempP(in), tempM(in);
845
846 //1. Apply plus and minus T and sum up
847 for( int j=1; j<(int)cphi; j++)
848 {
850 dg::blas2::symv( m_minus, helperP, tempP);
851 dg::blas1::axpby( (double)(cphi-j)/(double)cphi, tempP, 1., out );
852 helperP.swap(tempP);
854 dg::blas2::symv( m_plus, helperM, tempM);
855 dg::blas1::axpby( (double)(cphi-j)/(double)cphi, tempM, 1., out );
856 helperM.swap(tempM);
857 }
858 dg::blas1::scal( out, 1./(double)cphi);
859}

871void Fieldaligned<G, I, container>::zero( enum whichMatrix which,
872 const container& f, container& f0)
873{
874 dg::split( f, m_f, *m_g);
875 dg::split( f0, m_temp, *m_g);
876 //1. compute 2d interpolation in every plane and store in m_temp
877 for( unsigned i0=0; i0<m_Nz; i0++)
878 {
879 if(which == zeroPlus)
880 dg::blas2::symv( m_plus, m_f[i0], m_temp[i0]);
881 else if(which == zeroMinus)
882 dg::blas2::symv( m_minus, m_f[i0], m_temp[i0]);
883 else if(which == zeroPlusT)
884 {
885 if( ! m_have_adjoint) updateAdjoint( );

887 }
888 else if(which == zeroMinusT)
889 {
890 if( ! m_have_adjoint) updateAdjoint( );
891 dg::blas2::symv( m_minusT, m_f[i0], m_temp[i0]);
892 }

895template< class G, class I, class container>
896void Fieldaligned<G, I, container>::ePlus( enum whichMatrix which,
897 const container& f, container& fpe)
898{
899 dg::split( f, m_f, *m_g);
900 dg::split( fpe, m_temp, *m_g);
901 //1. compute 2d interpolation in every plane and store in m_temp
902 for( unsigned i0=0; i0<m_Nz; i0++)
903 {
904 unsigned ip = (i0==m_Nz-1) ? 0:i0+1;
905 if(which == einsPlus)
906 dg::blas2::symv( m_plus, m_f[ip], m_temp[i0]);
907 else if(which == einsMinusT)
908 {
909 if( ! m_have_adjoint) updateAdjoint( );
910 dg::blas2::symv( m_minusT, m_f[ip], m_temp[i0]);
911 }
912 }
913 //2. apply right boundary conditions in last plane
914 unsigned i0=m_Nz-1;
915 if( m_bcz != dg::PER)
916 {
917 if( m_bcz == dg::DIR || m_bcz == dg::NEU_DIR)
918 dg::blas1::axpby( 2, m_right, -1., m_f[i0], m_ghostP);
919 if( m_bcz == dg::NEU || m_bcz == dg::DIR_NEU)
920 dg::blas1::axpby( m_deltaPhi, m_right, 1., m_f[i0], m_ghostP);
921 //interlay ghostcells with periodic cells: L*g + (1-L)*fpe
922 dg::blas1::axpby( 1., m_ghostP, -1., m_temp[i0], m_ghostP);
923 dg::blas1::pointwiseDot( 1., m_limiter, m_ghostP, 1., m_temp[i0]);
924 }

929 const container& f, container& fme)
930{

932 dg::split( fme, m_temp, *m_g);
933 //1. compute 2d interpolation in every plane and store in m_temp
934 for( unsigned i0=0; i0<m_Nz; i0++)
935 {
936 unsigned im = (i0==0) ? m_Nz-1:i0-1;
937 if(which == einsPlusT)
938 {
939 if( ! m_have_adjoint) updateAdjoint( );
940 dg::blas2::symv( m_plusT, m_f[im], m_temp[i0]);
941 }
942 else if (which == einsMinus)
943 dg::blas2::symv( m_minus, m_f[im], m_temp[i0]);
944 }
945 //2. apply left boundary conditions in first plane
946 unsigned i0=0;
947 if( m_bcz != dg::PER)
948 {
949 if( m_bcz == dg::DIR || m_bcz == dg::DIR_NEU)
950 dg::blas1::axpby( 2., m_left, -1., m_f[i0], m_ghostM);
951 if( m_bcz == dg::NEU || m_bcz == dg::NEU_DIR)
952 dg::blas1::axpby( -m_deltaPhi, m_left, 1., m_f[i0], m_ghostM);
953 //interlay ghostcells with periodic cells: L*g + (1-L)*fme
954 dg::blas1::axpby( 1., m_ghostM, -1., m_temp[i0], m_ghostM);
955 dg::blas1::pointwiseDot( 1., m_limiter, m_ghostM, 1., m_temp[i0]);
956 }

959template<class G, class I, class container>
960template< class BinaryOp, class UnaryOp>
961container Fieldaligned<G, I,container>::evaluate( BinaryOp binary,
962 UnaryOp unary, unsigned p0, unsigned rounds) const
963{
964 //idea: simply apply I+/I- enough times on the init2d vector to get the result in each plane
965 //unary function is always such that the p0 plane is at x=0
966 assert( p0 < m_g->Nz());
967 const dg::ClonePtr<aGeometry2d> g2d = m_g->perp_grid();
968 container init2d = dg::pullback( binary, *g2d);
969 container zero2d = dg::evaluate( dg::zero, *g2d);
970
971 container temp(init2d), tempP(init2d), tempM(init2d);
972 container vec3d = dg::evaluate( dg::zero, *m_g);
973 std::vector<container> plus2d(m_Nz, zero2d), minus2d(plus2d), result(plus2d);
974 unsigned turns = rounds;
975 if( turns ==0) turns++;
976 //first apply Interpolation many times, scale and store results
977 for( unsigned r=0; r<turns; r++)
978 for( unsigned i0=0; i0<m_Nz; i0++)
979 {
980 dg::blas1::copy( init2d, tempP);
981 dg::blas1::copy( init2d, tempM);
982 unsigned rep = r*m_Nz + i0;
983 for(unsigned k=0; k<rep; k++)
984 {
986 dg::blas2::symv( m_minus, tempP, temp);
987 temp.swap( tempP);
989 dg::blas2::symv( m_plus, tempM, temp);
990 temp.swap( tempM);
991 }
992 dg::blas1::scal( tempP, unary( (double)rep*m_deltaPhi ) );
993 dg::blas1::scal( tempM, unary( -(double)rep*m_deltaPhi ) );
994 dg::blas1::axpby( 1., tempP, 1., plus2d[i0]);
995 dg::blas1::axpby( 1., tempM, 1., minus2d[i0]);
996 }
997 //now we have the plus and the minus filaments
998 if( rounds == 0) //there is a limiter
999 {
1000 for( unsigned i0=0; i0<m_Nz; i0++)
1001 {
1002 int idx = (int)i0 - (int)p0;
1003 if(idx>=0)
1004 result[i0] = plus2d[idx];
1005 else
1006 result[i0] = minus2d[abs(idx)];
1007 thrust::copy( result[i0].begin(), result[i0].end(), vec3d.begin() + i0*m_perp_size);
1008 }
1009 }
1010 else //sum up plus2d and minus2d
1011 {
1012 for( unsigned i0=0; i0<m_Nz; i0++)
1013 {
1014 unsigned revi0 = (m_Nz - i0)%m_Nz; //reverted index
1015 dg::blas1::axpby( 1., plus2d[i0], 0., result[i0]);
1016 dg::blas1::axpby( 1., minus2d[revi0], 1., result[i0]);
1017 }
1018 dg::blas1::axpby( -1., init2d, 1., result[0]);
1019 for(unsigned i0=0; i0<m_Nz; i0++)
1020 {
1021 int idx = ((int)i0 -(int)p0 + m_Nz)%m_Nz; //shift index
1022 thrust::copy( result[idx].begin(), result[idx].end(), vec3d.begin() + i0*m_perp_size);
1023 }
1024 }
1025 return vec3d;
1026}

1063thrust::host_vector<double> fieldaligned_evaluate(
1064 const aProductGeometry3d& grid,
1065 const CylindricalVectorLvl0& vec,
1066 const BinaryOp& binary,
1067 const UnaryOp& unary,
1068 unsigned p0,
1069 unsigned rounds,
1070 double eps = 1e-5)
1071{
1072 unsigned Nz = grid.Nz();
1073 const dg::ClonePtr<aGeometry2d> g2d = grid.perp_grid();
1074 // Construct for field-aligned output
1075 dg::HVec tempP = dg::evaluate( dg::zero, *g2d), tempM( tempP);
1076 std::vector<dg::HVec> plus2d(Nz, tempP), minus2d(plus2d), result(plus2d);

1080 dg::pullback( dg::cooX2d, *g2d),
1081 dg::pullback( dg::cooY2d, *g2d),
1082 dg::evaluate( dg::zero, *g2d)}, yy1(yy0), xx0( yy0), xx1(yy0); //s
1083 dg::geo::detail::DSFieldCylindrical3 cyl_field(vec);
1084 double deltaPhi = grid.hz();
1085 double phiM0 = 0., phiP0 = 0.;
1086 unsigned turns = rounds;
1087 if( turns == 0) turns++;
1088 for( unsigned r=0; r<turns; r++)
1089 for( unsigned i0=0; i0<Nz; i0++)
1090 {
1091 unsigned rep = r*Nz + i0;
1092 if( rep == 0)
1093 tempM = tempP = init2d;
1094 else
1095 {
1096 dg::Adaptive<dg::ERKStep<std::array<double,3>>> adapt(
1097 "Dormand-Prince-7-4-5", std::array<double,3>{0,0,0});
1098 dg::AdaptiveTimeloop<std::array<double,3>> odeint( adapt,
1099 cyl_field, dg::pid_control, dg::fast_l2norm, eps,
1100 1e-10);
1101 for( unsigned i=0; i<g2d->size(); i++)
1102 {
1103 // minus direction needs positive integration!
1104 double phiM1 = phiM0 + deltaPhi;

1106 coords0{yy0[0][i],yy0[1][i],yy0[2][i]}, coords1;
1107 odeint.integrate_in_domain( phiM0, coords0, phiM1, coords1,
1108 deltaPhi, *g2d, eps);
1109 yy1[0][i] = coords1[0], yy1[1][i] = coords1[1], yy1[2][i] =
1110 coords1[2];
1111 tempM[i] = binary( yy1[0][i], yy1[1][i]);
1112
1113 // plus direction needs negative integration!
1114 double phiP1 = phiP0 - deltaPhi;
1115 coords0 = std::array<double,3>{xx0[0][i],xx0[1][i],xx0[2][i]};
1116 odeint.integrate_in_domain( phiP0, coords0, phiP1, coords1,
1117 -deltaPhi, *g2d, eps);
1118 xx1[0][i] = coords1[0], xx1[1][i] = coords1[1], xx1[2][i] =
1119 coords1[2];
1120 tempP[i] = binary( xx1[0][i], xx1[1][i]);
1121 }
1122 std::swap( yy0, yy1);
1123 std::swap( xx0, xx1);
1124 phiM0 += deltaPhi;
1125 phiP0 -= deltaPhi;
1126 }
1127 dg::blas1::scal( tempM, unary( -(double)rep*deltaPhi ) );
1128 dg::blas1::scal( tempP, unary( (double)rep*deltaPhi ) );
1129 dg::blas1::axpby( 1., tempM, 1., minus2d[i0]);
1130 dg::blas1::axpby( 1., tempP, 1., plus2d[i0]);
1131 }
1132 //now we have the plus and the minus filaments
1133 if( rounds == 0) //there is a limiter
1134 {
1135 for( unsigned i0=0; i0<Nz; i0++)
1136 {
1137 int idx = (int)i0 - (int)p0;
1138 if(idx>=0)
1139 result[i0] = plus2d[idx];
1140 else
1141 result[i0] = minus2d[abs(idx)];
1142 thrust::copy( result[i0].begin(), result[i0].end(), vec3d.begin() +
1143 i0*g2d->size());
1144 }
1145 }
1146 else //sum up plus2d and minus2d
1147 {
1148 for( unsigned i0=0; i0<Nz; i0++)
1149 {
1150 unsigned revi0 = (Nz - i0)%Nz; //reverted index
1151 dg::blas1::axpby( 1., plus2d[i0], 0., result[i0]);
1152 dg::blas1::axpby( 1., minus2d[revi0], 1., result[i0]);
1153 }
1154 dg::blas1::axpby( -1., init2d, 1., result[0]);
1155 for(unsigned i0=0; i0<Nz; i0++)
1156 {
1157 int idx = ((int)i0 -(int)p0 + Nz)%Nz; //shift index
1158 thrust::copy( result[idx].begin(), result[idx].end(), vec3d.begin()
1159 + i0*g2d->size());
1160 }
1161 }
