631  std::vector<container> plus2d(m_Nz, zero2d), minus2d(plus2d), result(plus2d);
632  unsigned turns = rounds;
633  if( turns ==0) turns++;
634  //first apply Interpolation many times, scale and store results
635  for( unsigned r=0; r<turns; r++)
636  for( unsigned i0=0; i0<m_Nz; i0++)
637  {
638  dg::blas1::copy( init2d, tempP);
639  dg::blas1::copy( init2d, tempM);
640  unsigned rep = r*m_Nz + i0;
641  for(unsigned k=0; k<rep; k++)
642  {
643  dg::blas2::symv( m_plus, tempP, temp);
644  temp.swap( tempP);
645  dg::blas2::symv( m_minus, tempM, temp);
646  temp.swap( tempM);
647  }
648  dg::blas1::scal( tempP, unary( (double)rep*m_g.get().hz() ) );
649  dg::blas1::scal( tempM, unary( -(double)rep*m_g.get().hz() ) );
650  dg::blas1::axpby( 1., tempP, 1., plus2d[i0]);
651  dg::blas1::axpby( 1., tempM, 1., minus2d[i0]);
652  }
653  //now we have the plus and the minus filaments
654  if( rounds == 0) //there is a limiter
655  {
656  for( unsigned i0=0; i0<m_Nz; i0++)
657  {
658  int idx = (int)i0 - (int)p0;
659  if(idx>=0)
660  result[i0] = plus2d[idx];
661  else
662  result[i0] = minus2d[abs(idx)];
663  thrust::copy( result[i0].begin(), result[i0].end(), vec3d.begin() + i0*m_perp_size);
664  }
665  }
666  else //sum up plus2d and minus2d
667  {
668  for( unsigned i0=0; i0<m_Nz; i0++)
669  {
670  unsigned revi0 = (m_Nz - i0)%m_Nz; //reverted index
671  dg::blas1::axpby( 1., plus2d[i0], 0., result[i0]);
672  dg::blas1::axpby( 1., minus2d[revi0], 1., result[i0]);
673  }
674  dg::blas1::axpby( -1., init2d, 1., result[0]);
675  for(unsigned i0=0; i0<m_Nz; i0++)
676  {
677  int idx = ((int)i0 -(int)p0 + m_Nz)%m_Nz; //shift index
678  thrust::copy( result[idx].begin(), result[idx].end(), vec3d.begin() + i0*m_perp_size);
679  }
680  }
681  return vec3d;
682 }

692 template< class G, class I, class container>
693 void Fieldaligned<G, I, container>::ePlus( enum whichMatrix which, const container& f, container& fpe)
694 {
695  dg::split( f, m_f, m_g.get());
696  //1. compute 2d interpolation in every plane and store in m_temp
697  for( unsigned i0=0; i0<m_Nz; i0++)
698  {
699  unsigned ip = (i0==m_Nz-1) ? 0:i0+1;
700  if(which == einsPlus) dg::blas2::symv( m_plus, m_f[ip], m_temp[i0]);
701  else if(which == einsMinusT) dg::blas2::symv( m_minusT, m_f[ip], m_temp[i0]);
702  }
703  //2. apply right boundary conditions in last plane
704  unsigned i0=m_Nz-1;
705  if( m_bcz != dg::PER)
706  {
707  if( m_bcz == dg::DIR || m_bcz == dg::NEU_DIR)
708  dg::blas1::axpby( 2, m_right, -1., m_f[i0], m_ghostP);
709  if( m_bcz == dg::NEU || m_bcz == dg::DIR_NEU)
710  {
711  dg::blas1::pointwiseDot( m_right, m_hp, m_ghostP);
712  dg::blas1::axpby( 1., m_ghostP, 1., m_f[i0], m_ghostP);
713  }
714  //interlay ghostcells with periodic cells: L*g + (1-L)*fpe
715  dg::blas1::axpby( 1., m_ghostP, -1., m_temp[i0], m_ghostP);
716  dg::blas1::pointwiseDot( 1., m_limiter, m_ghostP, 1., m_temp[i0]);
717  }
718  dg::join( m_temp, fpe, m_g.get());
719 }

721 template< class G, class I, class container>
722 void Fieldaligned<G, I, container>::eMinus( enum whichMatrix which, const container& f, container& fme)
723 {
724  dg::split( f, m_f, m_g.get());
725  //1. compute 2d interpolation in every plane and store in m_temp
726  for( unsigned i0=0; i0<m_Nz; i0++)
727  {
728  unsigned im = (i0==0) ? m_Nz-1:i0-1;
729  if(which == einsPlusT) dg::blas2::symv( m_plusT, m_f[im], m_temp[i0]);
730  else if (which == einsMinus) dg::blas2::symv( m_minus, m_f[im], m_temp[i0]);
731  }
732  //2. apply left boundary conditions in first plane
733  unsigned i0=0;
734  if( m_bcz != dg::PER)
735  {
736  if( m_bcz == dg::DIR || m_bcz == dg::DIR_NEU)
737  dg::blas1::axpby( 2., m_left, -1., m_f[i0], m_ghostM);
738  if( m_bcz == dg::NEU || m_bcz == dg::NEU_DIR)
739  {
740  dg::blas1::pointwiseDot( m_left, m_hm, m_ghostM);
741  dg::blas1::axpby( -1., m_ghostM, 1., m_f[i0], m_ghostM);
742  }
743  //interlay ghostcells with periodic cells: L*g + (1-L)*fme
744  dg::blas1::axpby( 1., m_ghostM, -1., m_temp[i0], m_ghostM);
745  dg::blas1::pointwiseDot( 1., m_limiter, m_ghostM, 1., m_temp[i0]);
746  }
747  dg::join( m_temp, fme, m_g.get());
748 }
