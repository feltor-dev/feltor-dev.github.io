586 {

588  //unary function is always such that the p0 plane is at x=0
589  assert( p0 < m_g->Nz());

592  container zero2d = dg::evaluate( dg::zero, *g2d);
593

595  container vec3d = dg::evaluate( dg::zero, *m_g);
596  std::vector<container> plus2d(m_Nz, zero2d), minus2d(plus2d), result(plus2d);
597  unsigned turns = rounds;
598  if( turns ==0) turns++;
599  //first apply Interpolation many times, scale and store results
600  for( unsigned r=0; r<turns; r++)
601  for( unsigned i0=0; i0<m_Nz; i0++)
602  {
603  dg::blas1::copy( init2d, tempP);
604  dg::blas1::copy( init2d, tempM);
605  unsigned rep = r*m_Nz + i0;
606  for(unsigned k=0; k<rep; k++)
607  {
609  dg::blas2::symv( m_minus, tempP, temp);
610  temp.swap( tempP);
612  dg::blas2::symv( m_plus, tempM, temp);
613  temp.swap( tempM);
614  }
615  dg::blas1::scal( tempP, unary( (double)rep*m_g->hz() ) );
616  dg::blas1::scal( tempM, unary( -(double)rep*m_g->hz() ) );
617  dg::blas1::axpby( 1., tempP, 1., plus2d[i0]);
618  dg::blas1::axpby( 1., tempM, 1., minus2d[i0]);
619  }
620  //now we have the plus and the minus filaments
621  if( rounds == 0) //there is a limiter
622  {
623  for( unsigned i0=0; i0<m_Nz; i0++)
624  {
625  int idx = (int)i0 - (int)p0;
626  if(idx>=0)
627  result[i0] = plus2d[idx];
628  else
629  result[i0] = minus2d[abs(idx)];
630  thrust::copy( result[i0].begin(), result[i0].end(), vec3d.begin() + i0*m_perp_size);
631  }
632  }
633  else //sum up plus2d and minus2d
634  {
635  for( unsigned i0=0; i0<m_Nz; i0++)
636  {

638  dg::blas1::axpby( 1., plus2d[i0], 0., result[i0]);
639  dg::blas1::axpby( 1., minus2d[revi0], 1., result[i0]);
640  }

642  for(unsigned i0=0; i0<m_Nz; i0++)

651 template<class G, class I, class container>
652 container Fieldaligned<G, I,container>::interpolate_from_coarse_grid( const G& grid, const container& in)
653 {
654  //I think we need grid as input to split input vector and we need to interpret
655  //the grid nodes as node centered not cell-centered!
656  //idea: apply I+/I- cphi - 1 times in each direction and then apply interpolation formula
657  assert( m_g->Nz() % grid.Nz() == 0);
658  unsigned Nz_coarse = grid.Nz(), Nz = m_g->Nz();
659  unsigned cphi = Nz / Nz_coarse;
660 
661  container out = dg::evaluate( dg::zero, *m_g);
662  container helper = dg::evaluate( dg::zero, *m_g);
663  dg::split( helper, m_temp, *m_g);
664  std::vector<dg::View< container>> out_split = dg::split( out, *m_g);
665  std::vector<dg::View< const container>> in_split = dg::split( in, grid);
666  for ( int i=0; i<(int)Nz_coarse; i++)
667  {
668  //1. copy input vector to appropriate place in output
669  dg::blas1::copy( in_split[i], out_split[i*cphi]);
670  dg::blas1::copy( in_split[i], m_temp[i*cphi]);
671  }
672  //Step 1 needs to finish so that m_temp contains values everywhere
673  //2. Now apply plus and minus T to fill in the rest
674  for ( int i=0; i<(int)Nz_coarse; i++)
675  {
676  for( int j=1; j<(int)cphi; j++)
677  {
679  dg::blas2::symv( m_minus, out_split[i*cphi+j-1], out_split[i*cphi+j]);
681  dg::blas2::symv( m_plus, m_temp[(i*cphi+cphi+1-j)%Nz], m_temp[i*cphi+cphi-j]);
682  }
683  }
684  //3. Now add up with appropriate weights
685  for( int i=0; i<(int)Nz_coarse; i++)
686  for( int j=1; j<(int)cphi; j++)
687  {
688  double alpha = (double)(cphi-j)/(double)cphi;
689  double beta = (double)j/(double)cphi;
690  dg::blas1::axpby( alpha, out_split[i*cphi+j], beta, m_temp[i*cphi+j], out_split[i*cphi+j]);
691  }
692  return out;
693 }
694 template<class G, class I, class container>
695 void Fieldaligned<G, I,container>::integrate_between_coarse_grid( const G& grid, const container& in, container& out)
696 {
697  assert( m_g->Nz() % grid.Nz() == 0);
698  unsigned Nz_coarse = grid.Nz(), Nz = m_g->Nz();
699  unsigned cphi = Nz / Nz_coarse;
700 
701  out = in;
702  container helperP( in), helperM(in), tempP(in), tempM(in);
703 
704  //1. Apply plus and minus T and sum up
705  for( int j=1; j<(int)cphi; j++)
706  {
708  dg::blas2::symv( m_minus, helperP, tempP);
709  dg::blas1::axpby( (double)(cphi-j)/(double)cphi, tempP, 1., out );
710  helperP.swap(tempP);
712  dg::blas2::symv( m_plus, helperM, tempM);
713  dg::blas1::axpby( (double)(cphi-j)/(double)cphi, tempM, 1., out );
714  helperM.swap(tempM);
715  }
716  dg::blas1::scal( out, 1./(double)cphi);
717 }

726 template< class G, class I, class container>
727 void Fieldaligned<G, I, container>::ePlus( enum whichMatrix which, const container& f, container& fpe)
728 {
729  dg::split( f, m_f, *m_g);
730  dg::split( fpe, m_temp, *m_g);
731  //1. compute 2d interpolation in every plane and store in m_temp
732  for( unsigned i0=0; i0<m_Nz; i0++)
733  {
734  unsigned ip = (i0==m_Nz-1) ? 0:i0+1;
735  if(which == einsPlus) dg::blas2::symv( m_plus, m_f[ip], m_temp[i0]);
736  else if(which == einsMinusT) dg::blas2::symv( m_minusT, m_f[ip], m_temp[i0]);
737  }
738  //2. apply right boundary conditions in last plane
739  unsigned i0=m_Nz-1;
740  if( m_bcz != dg::PER)
741  {
742  if( m_bcz == dg::DIR || m_bcz == dg::NEU_DIR)
743  dg::blas1::axpby( 2, m_right, -1., m_f[i0], m_ghostP);
744  if( m_bcz == dg::NEU || m_bcz == dg::DIR_NEU)
745  {
746  dg::blas1::pointwiseDot( m_right, m_hp2d, m_ghostP);
747  dg::blas1::axpby( 1., m_ghostP, 1., m_f[i0], m_ghostP);
748  }
749  //interlay ghostcells with periodic cells: L*g + (1-L)*fpe
750  dg::blas1::axpby( 1., m_ghostP, -1., m_temp[i0], m_ghostP);
751  dg::blas1::pointwiseDot( 1., m_limiter, m_ghostP, 1., m_temp[i0]);
752  }
753 }

756 void Fieldaligned<G, I, container>::eMinus( enum whichMatrix which, const container& f, container& fme)
757 {
758  dg::split( f, m_f, *m_g);
759  dg::split( fme, m_temp, *m_g);

765  else if (which == einsMinus) dg::blas2::symv( m_minus, m_f[im], m_temp[i0]);
766  }
767  //2. apply left boundary conditions in first plane
768  unsigned i0=0;
769  if( m_bcz != dg::PER)
770  {
771  if( m_bcz == dg::DIR || m_bcz == dg::DIR_NEU)
772  dg::blas1::axpby( 2., m_left, -1., m_f[i0], m_ghostM);
773  if( m_bcz == dg::NEU || m_bcz == dg::NEU_DIR)
774  {
775  dg::blas1::pointwiseDot( m_left, m_hm2d, m_ghostM);
776  dg::blas1::axpby( -1., m_ghostM, 1., m_f[i0], m_ghostM);
777  }
778  //interlay ghostcells with periodic cells: L*g + (1-L)*fme
779  dg::blas1::axpby( 1., m_ghostM, -1., m_temp[i0], m_ghostM);
780  dg::blas1::pointwiseDot( 1., m_limiter, m_ghostM, 1., m_temp[i0]);
781  }
