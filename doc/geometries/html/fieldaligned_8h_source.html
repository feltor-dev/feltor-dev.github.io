475 container Fieldaligned<G, I,container>::evaluate( const BinaryOp& binary, const UnaryOp& unary, unsigned p0, unsigned rounds) const
476 {
477  //idea: simply apply I+/I- enough times on the init2d vector to get the result in each plane
478  //unary function is always such that the p0 plane is at x=0
479  assert( p0 < m_g->Nz());
480  const dg::ClonePtr<aGeometry2d> g2d = m_g->perp_grid();
481  container init2d = dg::pullback( binary, *g2d);
482  container zero2d = dg::evaluate( dg::zero, *g2d);
483 
484  container temp(init2d), tempP(init2d), tempM(init2d);
485  container vec3d = dg::evaluate( dg::zero, *m_g);
486  std::vector<container> plus2d(m_Nz, zero2d), minus2d(plus2d), result(plus2d);
487  unsigned turns = rounds;
488  if( turns ==0) turns++;
489  //first apply Interpolation many times, scale and store results
490  for( unsigned r=0; r<turns; r++)
491  for( unsigned i0=0; i0<m_Nz; i0++)
492  {
493  dg::blas1::copy( init2d, tempP);
494  dg::blas1::copy( init2d, tempM);
495  unsigned rep = r*m_Nz + i0;
496  for(unsigned k=0; k<rep; k++)
497  {
498  dg::blas2::symv( m_plus, tempP, temp);
499  temp.swap( tempP);
500  dg::blas2::symv( m_minus, tempM, temp);
501  temp.swap( tempM);
502  }
503  dg::blas1::scal( tempP, unary( (double)rep*m_g->hz() ) );
504  dg::blas1::scal( tempM, unary( -(double)rep*m_g->hz() ) );
505  dg::blas1::axpby( 1., tempP, 1., plus2d[i0]);
506  dg::blas1::axpby( 1., tempM, 1., minus2d[i0]);
507  }
508  //now we have the plus and the minus filaments
509  if( rounds == 0) //there is a limiter
510  {
511  for( unsigned i0=0; i0<m_Nz; i0++)
512  {
513  int idx = (int)i0 - (int)p0;
514  if(idx>=0)
515  result[i0] = plus2d[idx];
516  else
517  result[i0] = minus2d[abs(idx)];
518  thrust::copy( result[i0].begin(), result[i0].end(), vec3d.begin() + i0*m_perp_size);
519  }
520  }
521  else //sum up plus2d and minus2d
522  {
523  for( unsigned i0=0; i0<m_Nz; i0++)
524  {
525  unsigned revi0 = (m_Nz - i0)%m_Nz; //reverted index
526  dg::blas1::axpby( 1., plus2d[i0], 0., result[i0]);
527  dg::blas1::axpby( 1., minus2d[revi0], 1., result[i0]);
528  }
529  dg::blas1::axpby( -1., init2d, 1., result[0]);
530  for(unsigned i0=0; i0<m_Nz; i0++)
531  {
532  int idx = ((int)i0 -(int)p0 + m_Nz)%m_Nz; //shift index
533  thrust::copy( result[idx].begin(), result[idx].end(), vec3d.begin() + i0*m_perp_size);
534  }
535  }
536  return vec3d;
537 }

547 template< class G, class I, class container>
548 void Fieldaligned<G, I, container>::ePlus( enum whichMatrix which, const container& f, container& fpe)
549 {
550  dg::split( f, m_f, *m_g);
551  dg::split( fpe, m_temp, *m_g);
552  //1. compute 2d interpolation in every plane and store in m_temp
553  for( unsigned i0=0; i0<m_Nz; i0++)
554  {
555  unsigned ip = (i0==m_Nz-1) ? 0:i0+1;
556  if(which == einsPlus) dg::blas2::symv( m_plus, m_f[ip], m_temp[i0]);
557  else if(which == einsMinusT) dg::blas2::symv( m_minusT, m_f[ip], m_temp[i0]);
558  }
559  //2. apply right boundary conditions in last plane
560  unsigned i0=m_Nz-1;
561  if( m_bcz != dg::PER)
562  {
563  if( m_bcz == dg::DIR || m_bcz == dg::NEU_DIR)
564  dg::blas1::axpby( 2, m_right, -1., m_f[i0], m_ghostP);
565  if( m_bcz == dg::NEU || m_bcz == dg::DIR_NEU)
566  {
567  dg::blas1::pointwiseDot( m_right, m_hp, m_ghostP);
568  dg::blas1::axpby( 1., m_ghostP, 1., m_f[i0], m_ghostP);
569  }
570  //interlay ghostcells with periodic cells: L*g + (1-L)*fpe
571  dg::blas1::axpby( 1., m_ghostP, -1., m_temp[i0], m_ghostP);
572  dg::blas1::pointwiseDot( 1., m_limiter, m_ghostP, 1., m_temp[i0]);
573  }
