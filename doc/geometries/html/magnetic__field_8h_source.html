909  for( auto h : m_horizontal)
910  h_dist.push_back( Z-h );
911  double v_min = *std::min_element( v_dist.begin(), v_dist.end(),
912  [](double a, double b){ return fabs(a) < fabs(b);} );
913  double h_min = *std::min_element( h_dist.begin(), h_dist.end(),
914  [](double a, double b){ return fabs(a) < fabs(b);} );
915  if( fabs(v_min) < fabs(h_min) ) // if vertical R wall is closer
916  {
917  double br = m_BR( R,Z);
918  return v_min*br < 0 ? +1 : -1;
919  }
920  else //horizontal Z wall is closer
921  {
922  double bz = m_BZ( R,Z);
923  return h_min*bz < 0 ? +1 : -1;
924  }
925  }
926  private:
927  std::vector<double> m_vertical, m_horizontal;
