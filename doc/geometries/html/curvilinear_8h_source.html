50 void square( const dg::SparseTensor<thrust::host_vector<double> >& jac, const thrust::host_vector<double>& R, dg::SparseTensor<thrust::host_vector<double> >& metric, bool orthogonal)
51 {
52  thrust::host_vector<double> tempxx( R.size()), tempxy(R.size()), tempyy(R.size()), temppp(R.size());
53  for( unsigned i=0; i<R.size(); i++)
54  {
55  tempxx[i] = (jac.value(0,0)[i]*jac.value(0,0)[i]+jac.value(0,1)[i]*jac.value(0,1)[i]);
56  tempxy[i] = (jac.value(0,0)[i]*jac.value(1,0)[i]+jac.value(0,1)[i]*jac.value(1,1)[i]);
57  tempyy[i] = (jac.value(1,0)[i]*jac.value(1,0)[i]+jac.value(1,1)[i]*jac.value(1,1)[i]);
58  temppp[i] = 1./R[i]/R[i]; //1/R^2
59  }
60  metric.idx(0,0) = 0; metric.value(0) = tempxx;
61  metric.idx(1,1) = 1; metric.value(1) = tempyy;
62  metric.idx(2,2) = 2; metric.value(2) = temppp;
63  if( !orthogonal)
64  {
65  metric.idx(0,1) = metric.idx(1,0) = 3;
66  metric.value(3) = tempxy;
67  }
68 }
