53 {

58  dg::blas1::pointwiseDot( 1., jac.value(1,0), jac.value(1,0), 1., jac.value(1,1), jac.value(1,1), 0., values[2]); //yy
59  dg::blas1::pointwiseDot( values[3], values[3], values[3]);
60  dg::blas1::pointwiseDivide( 1., values[3], values[3]); //pp == 1/R^2
61

63  }
64  SparseTensor<host_vector> metric(values[0]); //unit tensor
65  metric.values().pop_back(); //remove the one
66  metric.idx(0,0) = 1; metric.values().push_back( values[1]);
67  metric.idx(1,1) = 2; metric.values().push_back( values[2]);
68  metric.idx(2,2) = 3; metric.values().push_back( values[3]);
69  if( !orthogonal)
70  {
71  metric.idx(1,0) = metric.idx(0,1) = 4;
72  metric.values().push_back( values[4]);
73  }
