49template<class host_vector>
50dg::SparseTensor<host_vector> square( const dg::SparseTensor<host_vector >& jac, const host_vector& R, bool orthogonal)
51{
52 std::vector<host_vector> values( 5, R);
53 {
54 dg::blas1::scal( values[0], 0); //0
55 dg::blas1::pointwiseDot( 1., jac.value(0,0), jac.value(0,0), 1., jac.value(0,1), jac.value(0,1), 0., values[1]); //xx
56 dg::blas1::pointwiseDot( 1., jac.value(1,0), jac.value(1,0), 1., jac.value(1,1), jac.value(1,1), 0., values[2]); //yy
57 dg::blas1::pointwiseDot( values[3], values[3], values[3]);
58 dg::blas1::pointwiseDivide( 1., values[3], values[3]); //pp == 1/R^2
59
60 dg::blas1::pointwiseDot( 1., jac.value(0,0), jac.value(1,0), 1., jac.value(0,1), jac.value(1,1), 0., values[4]); //xy
61 }
62 SparseTensor<host_vector> metric(values[0]); //unit tensor
63 metric.values().pop_back(); //remove the one
64 metric.idx(0,0) = 1; metric.values().push_back( values[1]);
65 metric.idx(1,1) = 2; metric.values().push_back( values[2]);
66 metric.idx(2,2) = 3; metric.values().push_back( values[3]);
67 if( !orthogonal)
68 {
69 metric.idx(1,0) = metric.idx(0,1) = 4;
70 metric.values().push_back( values[4]);
71 }
72 return metric;
73}
