472  dg::blas1::pointwiseDivide( zetaU, temp_scalar, zetaU);
473  dg::blas1::pointwiseDivide( etaU, temp_scalar, etaU);
474  //now compute etaV and its inverse
475  container etaVinv(u_zeta), etaV(etaVinv);
476  dg::blas1::pointwiseDot( etaVinv, chi.value(0,0), etaVinv);
477  container perp_vol = dg::tensor::volume(g2d_.metric());
478  dg::blas1::pointwiseDot( etaVinv, perp_vol, etaVinv);
479  dg::blas1::transform( etaVinv, etaV, dg::INVERT<double>());
480  thrust::host_vector<double> etaVinv_h;
481  dg::blas1::transfer( etaVinv, etaVinv_h);
482  //now compute v_zeta and v_eta
483  container v_zeta(u), v_eta(u);
484  dg::blas1::axpby( -1., temp_eta, 0.,v_zeta);
485  dg::blas1::axpby( +1., temp_zeta, 0.,v_eta);
486  dg::blas1::pointwiseDot( v_zeta, perp_vol, v_zeta);
487  dg::blas1::pointwiseDot( v_eta, perp_vol, v_eta);
488 
489  //construct c0 and scale all vector components with it
490  c0_ = fabs( detail::construct_c0( etaVinv_h, g2d_));
491  if( psi1 < psi0) c0_*=-1;
492  lu_ = c0_*(psi1-psi0);
493  //std::cout << "c0 is "<<c0_<<"\n";
494  dg::blas1::scal( etaV, 1./c0_);
495  dg::blas1::scal( zetaU, 1./c0_);
496  dg::blas1::scal( etaU, 1./c0_);
497  dg::blas1::scal( u_zeta, c0_);
498  dg::blas1::scal( v_zeta, c0_);
499  dg::blas1::scal( u_eta, c0_);
500  dg::blas1::scal( v_eta, c0_);
501  //transfer to host
502  detail::transform( u_zeta, u_eta, ux_, uy_, g2d_);
503  detail::transform( v_zeta, v_eta, vx_, vy_, g2d_);
504  dg::assign( etaV, etaV_);
505  dg::assign( etaU, etaU_);
506  dg::assign( zetaU, zetaU_);
507  }

510  double c0_, lu_;
511  thrust::host_vector<double> u_, ux_, uy_, vx_, vy_;
