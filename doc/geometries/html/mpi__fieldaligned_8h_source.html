259 template< class BinaryOp, class UnaryOp>
260 MPI_Vector<container> Fieldaligned<G,MPIDistMat<M,C>, MPI_Vector<container> >::evaluate( BinaryOp binary, UnaryOp unary, unsigned p0, unsigned rounds) const
261 {
262  //idea: simply apply I+/I- enough times on the init2d vector to get the result in each plane
263  //unary function is always such that the p0 plane is at x=0
264  assert( p0 < m_g.get().global().Nz());
265  const dg::ClonePtr<aMPIGeometry2d> g2d = m_g.get().perp_grid();
266  MPI_Vector<container> init2d = dg::pullback( binary, g2d.get());
267  MPI_Vector<container> zero2d = dg::evaluate( dg::zero, g2d.get());
268  unsigned globalNz = m_g.get().global().Nz();
269 
270  MPI_Vector<container> temp(init2d), tempP(init2d), tempM(init2d);
271  MPI_Vector<container> vec3d = dg::evaluate( dg::zero, m_g.get());
272  std::vector<MPI_Vector<container> > plus2d(globalNz, zero2d), minus2d(plus2d), result(plus2d);
273  unsigned turns = rounds;
274  if( turns ==0) turns++;
275  //first apply Interpolation many times, scale and store results
276  for( unsigned r=0; r<turns; r++)
277  for( unsigned i0=0; i0<globalNz; i0++)
278  {
279  dg::blas1::copy( init2d, tempP);
280  dg::blas1::copy( init2d, tempM);
281  unsigned rep = r*globalNz + i0;
282  for(unsigned k=0; k<rep; k++)
283  {
284  dg::blas2::symv( m_plus, tempP, temp);
285  temp.swap( tempP);
286  dg::blas2::symv( m_minus, tempM, temp);
287  temp.swap( tempM);
288  }
289  dg::blas1::scal( tempP, unary( (double)rep*m_g.get().hz() ) );
290  dg::blas1::scal( tempM, unary( -(double)rep*m_g.get().hz() ) );
291  dg::blas1::axpby( 1., tempP, 1., plus2d[i0]);
292  dg::blas1::axpby( 1., tempM, 1., minus2d[i0]);
293  }
294  //now we have the plus and the minus filaments
295  if( rounds == 0) //there is a limiter
296  {
297  for( unsigned i0=0; i0<m_Nz; i0++)
298  {
299  int idx = (int)(i0+m_coords2*m_Nz) - (int)p0;
300  if(idx>=0)
301  result[i0] = plus2d[idx];
302  else
303  result[i0] = minus2d[abs(idx)];
304  thrust::copy( result[i0].data().begin(), result[i0].data().end(), vec3d.data().begin() + i0*m_perp_size);
305  }
306  }
307  else //sum up plus2d and minus2d
308  {
309  for( unsigned i0=0; i0<globalNz; i0++)
310  {
311  unsigned revi0 = (globalNz - i0)%globalNz; //reverted index
312  dg::blas1::axpby( 1., plus2d[i0], 0., result[i0]);
313  dg::blas1::axpby( 1., minus2d[revi0], 1., result[i0]);
314  }
315  dg::blas1::axpby( -1., init2d, 1., result[0]);
316  for(unsigned i0=0; i0<m_Nz; i0++)
317  {
318  int idx = ((int)i0 + m_coords2*m_Nz -(int)p0 + globalNz)%globalNz; //shift index
319  thrust::copy( result[idx].data().begin(), result[idx].data().end(), vec3d.data().begin() + i0*m_perp_size);
320  }
321  }
322  return vec3d;
323 }

333 void Fieldaligned<G,MPIDistMat<M,C>, MPI_Vector<container> >::ePlus( enum whichMatrix which, const MPI_Vector<container>& f, MPI_Vector<container>& fpe )
334 {
335  dg::split( f, m_f, m_g.get());
336  //1. compute 2d interpolation in every plane and store in m_temp
337  for( unsigned i0=0; i0<m_Nz; i0++)
338  {
339  unsigned ip = (i0==m_Nz-1) ? 0:i0+1;
340  if(which == einsPlus) dg::blas2::symv( m_plus, m_f[ip], m_temp[i0]);
341  else if(which == einsMinusT) dg::blas2::symv( m_minusT, m_f[ip], m_temp[i0]);
342  }
343 
344  //2. communicate halo in z
345  if( m_sizeZ != 1)
346  {
347  unsigned i0 = m_Nz-1;
348  detail::sendBackward( m_temp[i0].data(), m_ghostM.data(), m_g.get().communicator());
349  m_temp[i0].swap( m_ghostM);
350  }
351 
352  //3. apply right boundary conditions in last plane
353  unsigned i0=m_Nz-1;
354  if( m_bcz != dg::PER && m_g.get().z1() == m_g.get().global().z1())
355  {
356  if( m_bcz == dg::DIR || m_bcz == dg::NEU_DIR)
357  dg::blas1::axpby( 2, m_right, -1., m_f[i0], m_ghostP);
358  if( m_bcz == dg::NEU || m_bcz == dg::DIR_NEU)
359  {
360  dg::blas1::pointwiseDot( m_right, m_hp, m_ghostP);
361  dg::blas1::axpby( 1., m_ghostP, 1., m_f[i0], m_ghostP);
362  }
363  //interlay ghostcells with periodic cells: L*g + (1-L)*fpe
364  dg::blas1::axpby( 1., m_ghostP, -1., m_temp[i0], m_ghostP);
365  dg::blas1::pointwiseDot( 1., m_limiter, m_ghostP, 1., m_temp[i0]);
366  }
367  dg::join( m_temp, fpe, m_g.get());
368 }
