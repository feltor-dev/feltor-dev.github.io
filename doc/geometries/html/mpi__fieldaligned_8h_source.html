318 template< class BinaryOp, class UnaryOp>
319 MPI_Vector<container> Fieldaligned<G,MPIDistMat<M,C>, MPI_Vector<container> >::evaluate( BinaryOp binary, UnaryOp unary, unsigned p0, unsigned rounds) const
320 {
321  //idea: simply apply I+/I- enough times on the init2d vector to get the result in each plane
322  //unary function is always such that the p0 plane is at x=0
323  assert( p0 < m_g->global().Nz());
324  const dg::ClonePtr<aMPIGeometry2d> g2d = m_g->perp_grid();
325  MPI_Vector<container> init2d = dg::pullback( binary, *g2d);
326  MPI_Vector<container> zero2d = dg::evaluate( dg::zero, *g2d);
327  unsigned globalNz = m_g->global().Nz();
328 
329  MPI_Vector<container> temp(init2d), tempP(init2d), tempM(init2d);
330  MPI_Vector<container> vec3d = dg::evaluate( dg::zero, *m_g);
331  std::vector<MPI_Vector<container> > plus2d(globalNz, zero2d), minus2d(plus2d), result(plus2d);
332  unsigned turns = rounds;
333  if( turns ==0) turns++;
334  //first apply Interpolation many times, scale and store results
335  for( unsigned r=0; r<turns; r++)
336  for( unsigned i0=0; i0<globalNz; i0++)
337  {
338  dg::blas1::copy( init2d, tempP);
339  dg::blas1::copy( init2d, tempM);
340  unsigned rep = r*globalNz + i0;
341  for(unsigned k=0; k<rep; k++)
342  {
344  dg::blas2::symv( m_minus, tempP, temp);
345  temp.swap( tempP);
347  dg::blas2::symv( m_plus, tempM, temp);
348  temp.swap( tempM);
349  }
350  dg::blas1::scal( tempP, unary( (double)rep*m_g->hz() ) );
351  dg::blas1::scal( tempM, unary( -(double)rep*m_g->hz() ) );
352  dg::blas1::axpby( 1., tempP, 1., plus2d[i0]);
353  dg::blas1::axpby( 1., tempM, 1., minus2d[i0]);
354  }
355  //now we have the plus and the minus filaments
356  if( rounds == 0) //there is a limiter
357  {
358  for( unsigned i0=0; i0<m_Nz; i0++)
359  {
360  int idx = (int)(i0+m_coords2*m_Nz) - (int)p0;
361  if(idx>=0)
362  result[i0] = plus2d[idx];
363  else
364  result[i0] = minus2d[abs(idx)];
365  thrust::copy( result[i0].data().begin(), result[i0].data().end(), vec3d.data().begin() + i0*m_perp_size);
366  }
367  }
368  else //sum up plus2d and minus2d
369  {
370  for( unsigned i0=0; i0<globalNz; i0++)
371  {
372  unsigned revi0 = (globalNz - i0)%globalNz; //reverted index
373  dg::blas1::axpby( 1., plus2d[i0], 0., result[i0]);
374  dg::blas1::axpby( 1., minus2d[revi0], 1., result[i0]);
375  }
376  dg::blas1::axpby( -1., init2d, 1., result[0]);
377  for(unsigned i0=0; i0<m_Nz; i0++)
378  {
379  int idx = ((int)i0 + m_coords2*m_Nz -(int)p0 + globalNz)%globalNz; //shift index
380  thrust::copy( result[idx].data().begin(), result[idx].data().end(), vec3d.data().begin() + i0*m_perp_size);
381  }
382  }
383  return vec3d;
384 }

393 template<class G, class M, class C, class container>
394 void Fieldaligned<G,MPIDistMat<M,C>, MPI_Vector<container> >::ePlus( enum whichMatrix which, const MPI_Vector<container>& f, MPI_Vector<container>& fpe )
395 {
396  dg::split( f, m_f, *m_g);
397  dg::split( fpe, m_temp, *m_g);
398  //1. compute 2d interpolation in every plane and store in m_temp
399  for( unsigned i0=0; i0<m_Nz; i0++)
400  {
401  unsigned ip = (i0==m_Nz-1) ? 0:i0+1;
402  if(which == einsPlus) dg::blas2::symv( m_plus, m_f[ip], m_temp[i0]);
403  else if(which == einsMinusT) dg::blas2::symv( m_minusT, m_f[ip], m_temp[i0]);
404  }
405 
406  //2. communicate halo in z
407  if( m_sizeZ != 1)
408  {
409  unsigned i0 = m_Nz-1;
410 #ifdef _DG_CUDA_UNAWARE_MPI
411  thrust::copy( m_temp[i0].data().cbegin(), m_temp[i0].data().cend(), m_send_buffer.begin());
412  detail::sendBackward( m_send_buffer, m_recv_buffer, m_g->communicator());
413  thrust::copy( m_recv_buffer.cbegin(), m_recv_buffer.cend(), m_temp[i0].data().begin());
414 #else
415  detail::sendBackward( m_temp[i0].data(), m_ghostM.data(), m_g->communicator());
416  dg::blas1::copy( m_ghostM, m_temp[i0]);
417 #endif //_DG_CUDA_UNAWARE_MPI
418  }
419 
420  //3. apply right boundary conditions in last plane
421  unsigned i0=m_Nz-1;
422  if( m_bcz != dg::PER && m_g->local().z1() == m_g->global().z1())
423  {
424  if( m_bcz == dg::DIR || m_bcz == dg::NEU_DIR)
425  dg::blas1::axpby( 2, m_right, -1., m_f[i0], m_ghostP);
426  if( m_bcz == dg::NEU || m_bcz == dg::DIR_NEU)
427  {
428  dg::blas1::pointwiseDot( m_right, m_hp2d, m_ghostP);
429  dg::blas1::axpby( 1., m_ghostP, 1., m_f[i0], m_ghostP);
430  }
431  //interlay ghostcells with periodic cells: L*g + (1-L)*fpe
432  dg::blas1::axpby( 1., m_ghostP, -1., m_temp[i0], m_ghostP);
433  dg::blas1::pointwiseDot( 1., m_limiter, m_ghostP, 1., m_temp[i0]);
434  }
435 }
