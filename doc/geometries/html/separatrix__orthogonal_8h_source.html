22//compute the vector of r and z - values that form one psi surface
23//assumes y_0 = 0
24template<class real_type>
25void computeX_rzy( const CylindricalFunctorsLvl1& psi,
26 const thrust::host_vector<real_type>& y_vec,
27 const unsigned nodeX0, const unsigned nodeX1,
28 thrust::host_vector<real_type>& r, //output r - values
29 thrust::host_vector<real_type>& z, //output z - values
30 const real_type* R_init, const real_type* Z_init, //2 input coords on perp line
31 real_type f_psi, //input f
32 int mode, bool verbose = false )
33{
34 thrust::host_vector<real_type> r_old(y_vec.size(), 0), r_diff( r_old);
35 thrust::host_vector<real_type> z_old(y_vec.size(), 0), z_diff( z_old);
36 r.resize( y_vec.size()), z.resize(y_vec.size());
37 std::array<real_type,2> begin{ {0,0} }, end(begin), temp(begin);
38 begin[0] = R_init[0], begin[1] = Z_init[0];
39 dg::geo::ribeiro::FieldRZY fieldRZYconf(psi);
40 dg::geo::equalarc::FieldRZY fieldRZYequi(psi);
41 fieldRZYconf.set_f(f_psi);
42 fieldRZYequi.set_f(f_psi);
43 unsigned steps = 1; real_type eps = 1e10, eps_old=2e10;
44 using Vec = std::array<real_type,2>;
45 dg::SinglestepTimeloop<Vec> odeint;
46 if( mode == 0)
47 odeint.construct( dg::RungeKutta<Vec>( "Feagin-17-8-10", begin),
48 fieldRZYconf);
49 if(mode == 1)
50 odeint.construct( dg::RungeKutta<Vec>( "Feagin-17-8-10", begin),
51 fieldRZYequi);
52 while( (eps < eps_old||eps > 1e-7) && eps > 1e-11)
53 {
54 eps_old = eps, r_old = r, z_old = z;
56 if( nodeX0 != 0)
57 {
58 begin[0] = R_init[1], begin[1] = Z_init[1];
59 odeint.integrate_steps( 0, begin, y_vec[nodeX0-1], end, steps);
60 r[nodeX0-1] = end[0], z[nodeX0-1] = end[1];
61 }
62 for( int i=nodeX0-2; i>=0; i--)
63 {
64 temp = end;
65 odeint.integrate_steps( y_vec[i+1], temp, y_vec[i], end, steps);
66 r[i] = end[0], z[i] = end[1];
67 }
69 begin[0] = R_init[0], begin[1] = Z_init[0];
70 odeint.integrate_steps( 0, begin, y_vec[nodeX0], end, steps);
71 r[nodeX0] = end[0], z[nodeX0] = end[1];
72 for( unsigned i=nodeX0+1; i<nodeX1; i++)
73 {
74 temp = end;
75 odeint.integrate_steps( y_vec[i-1], temp, y_vec[i], end, steps);
76 r[i] = end[0], z[i] = end[1];
77 }
78 temp = end;
79 odeint.integrate_steps( y_vec[nodeX1-1], temp, 2.*M_PI, end, steps);
80 eps = sqrt( (end[0]-R_init[0])*(end[0]-R_init[0]) + (end[1]-Z_init[0])*(end[1]-Z_init[0]));
81 if(verbose)std::cout << "abs. error is "<<eps<<" with "<<steps<<" steps\n";
83 if( nodeX0!= 0)
84 {
85 begin[0] = R_init[1], begin[1] = Z_init[1];
86 odeint.integrate_steps( 2.*M_PI, begin, y_vec[nodeX1], end, steps);
87 r[nodeX1] = end[0], z[nodeX1] = end[1];
88 }
89 for( unsigned i=nodeX1+1; i<y_vec.size(); i++)
90 {
91 temp = end;
92 odeint.integrate_steps( y_vec[i-1], temp, y_vec[i], end, steps);
93 r[i] = end[0], z[i] = end[1];
94 }
95 //compute error in R,Z only
96 dg::blas1::axpby( 1., r, -1., r_old, r_diff);
97 dg::blas1::axpby( 1., z, -1., z_old, z_diff);
98 real_type er = dg::blas1::dot( r_diff, r_diff);
99 real_type ez = dg::blas1::dot( z_diff, z_diff);
100 real_type ar = dg::blas1::dot( r, r);
101 real_type az = dg::blas1::dot( z, z);
102 eps = sqrt( er + ez)/sqrt(ar+az);
103 if(verbose)std::cout << "rel. error is "<<eps<<" with "<<steps<<" steps\n";
104 steps*=2;
105 }
106 r = r_old, z = z_old;
107}
