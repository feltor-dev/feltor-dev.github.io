301  const dg::geo::CylindricalVectorLvl0& bhat, const Geometry3d& g)
302 {
303  using host_vector = dg::get_host_vector<Geometry3d>;
304  SparseTensor<host_vector> t;
305  std::array<host_vector,3> bt;
306  dg::pushForward( bhat.x(), bhat.y(), bhat.z(), bt[0], bt[1], bt[2], g);
307  std::vector<host_vector> chi(6, dg::evaluate( dg::zero,g));
308  dg::blas1::pointwiseDot( bt[0], bt[0], chi[0]);
309  dg::blas1::pointwiseDot( bt[0], bt[1], chi[1]);
310  dg::blas1::pointwiseDot( bt[0], bt[2], chi[2]);
311  dg::blas1::pointwiseDot( bt[1], bt[1], chi[3]);
312  dg::blas1::pointwiseDot( bt[1], bt[2], chi[4]);
313  dg::blas1::pointwiseDot( bt[2], bt[2], chi[5]);
314  t.idx(0,0) = 0, t.idx(0,1) = t.idx(1,0) = 1,
315  t.idx(0,2) = t.idx(2,0) = 2;
316  t.idx(1,1) = 3, t.idx(1,2) = t.idx(2,1) = 4;
317  t.idx(2,2) = 5;
318  t.values() = chi;
319  return t;
320 }
