97 void transform_psi( TokamakMagneticField mag, double& psi0, double& alpha0, double& sign0)
98 {
99  double RO=mag.R0(), ZO=0.;
100  dg::geo::findOpoint( mag.get_psip(), RO, ZO);
101  double psipO = mag.psip()( RO, ZO);
102  double wall_psi0p = (1.-psi0*psi0)*psipO;
103  double wall_alpha0p = -(2.*psi0+alpha0)*alpha0*psipO;
104  psi0 = wall_psi0p + sign0*wall_alpha0p/2.;
105  alpha0 = fabs( wall_alpha0p/2.);
106  sign0 = sign0*((psipO>0)-(psipO<0));
107 }

321 static inline void createSheathRegion(
322  Json::Value jsmod, dg::file::error mode, TokamakMagneticField mag,
323  CylindricalFunctor wall, double R0, double R1, double Z0, double Z1,
324  CylindricalFunctor& sheath, CylindricalFunctor& direction )
325 {
326  Grid1d gR1d ( R0, R1, 1, 100);
327  Grid1d gZ1d ( Z0, Z1, 1, 100);
328  std::array<bool,2> sheathR = {false,false}, sheathZ = {false,false};
329  for ( unsigned i=0; i<100; i++)
330  {
331  if( wall( R0+i*gR1d.h(), Z0) == 0)
332  sheathR[0] = true;
333  if( wall( R0+i*gR1d.h(), Z1) == 0)
334  sheathR[1] = true;
335  if( wall( R0, Z0 + i*gZ1d.h()) == 0)
336  sheathZ[0] = true;
337  if( wall( R1, Z0 + i*gZ1d.h()) == 0)
338  sheathZ[1] = true;
339  }
340  std::vector<double> horizontal_sheath, vertical_sheath;
341  if( true == sheathR[0])
342  horizontal_sheath.push_back( Z0);
343  if( true == sheathR[1])
344  horizontal_sheath.push_back( Z1);
345  if( true == sheathZ[0])
346  vertical_sheath.push_back( R0);
347  if( true == sheathZ[1])
348  vertical_sheath.push_back( R1);
349  //direction
350  direction = dg::geo::WallDirection( mag, vertical_sheath, horizontal_sheath);
351  //sheath
352  CylindricalFunctor dist = dg::WallDistance( vertical_sheath, horizontal_sheath);
353  double boundary = dg::file::get( mode, jsmod, "sheath", "boundary", 0.1 ).asDouble();
354  double alpha = dg::file::get( mode, jsmod, "sheath", "alpha", 0.01 ).asDouble();
355  double a = mag.params().a();
356  dg::PolynomialHeaviside poly( boundary*a - alpha*a/2., alpha*a/2., -1);
357  sheath = dg::compose( poly, dist);
358  sheath = mod::SetIntersection( mod::SetNot( wall), sheath);
359 }
