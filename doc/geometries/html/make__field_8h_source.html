74static inline TokamakMagneticField createMagneticField( dg::file::WrappedJsonValue gs)
75{
76 std::string e = gs.get( "equilibrium", "solovev" ).asString();
77 equilibrium equi = str2equilibrium.at( e);
78 switch( equi){
79 case equilibrium::polynomial:
80 {
81 polynomial::Parameters gp( gs);
82 return createPolynomialField( gp);
83 }
84 case equilibrium::toroidal:
85 {
86 double R0 = gs.get( "R_0", 10.0).asDouble();
87 return createToroidalField( R0);
88 }
89 case equilibrium::guenther:
90 {
91 double I0 = gs.get( "I_0", 20.0).asDouble();
92 double R0 = gs.get( "R_0", 10.0).asDouble();
93 return createGuentherField( R0, I0);
94 }
95 case equilibrium::circular:
96 {
97 double I0 = gs.get( "I_0", 20.0).asDouble();
98 double R0 = gs.get( "R_0", 10.0).asDouble();
99 return createCircularField( R0, I0);
100 }
101#ifdef BOOST_VERSION
102 case equilibrium::taylor:
103 {
104 solovev::Parameters gp( gs);
105 return createTaylorField( gp);
106 }
107#endif
108 default:
109 {
110 solovev::Parameters gp( gs);
111 return createSolovevField( gp);
112 }
113 }

119void transform_psi( TokamakMagneticField mag, double& psi0, double& alpha0, double& sign0)
120{
121 double RO=mag.R0(), ZO=0.;
122 dg::geo::findOpoint( mag.get_psip(), RO, ZO);
123 double psipO = mag.psip()( RO, ZO);
124 double wall_psi0p = (1.-psi0*psi0)*psipO;
125 double wall_alpha0p = -(2.*psi0+alpha0)*alpha0*psipO;
126 psi0 = wall_psi0p + sign0*wall_alpha0p/2.;
127 alpha0 = fabs( wall_alpha0p/2.);
128 sign0 = sign0*((psipO>0)-(psipO<0));
129}

169static inline TokamakMagneticField createModifiedField(
170 dg::file::WrappedJsonValue gs, dg::file::WrappedJsonValue jsmod,
171 CylindricalFunctor& wall, CylindricalFunctor& transition)
172{
173 TokamakMagneticField mag = createMagneticField( gs);
174 const MagneticFieldParameters& inp = mag.params();
175 description desc = inp.getDescription();
176 equilibrium equi = inp.getEquilibrium();
177 std::string m = jsmod.get( "type", "heaviside" ).asString();
178 modifier mod = str2modifier.at( m);
179 MagneticFieldParameters mod_params{ inp.a(), inp.elongation(),
180 inp.triangularity(), equi, mod, desc};
181 CylindricalFunctorsLvl2 mod_psip;
182 switch (mod) {
183 default: //none
184 {
185 wall = mod::DampingRegion( mod::nowhere, mag.psip(), 0, 1, -1);
186 transition = mod::MagneticTransition( mod::nowhere, mag.psip(),
187 0, 1, -1);
188 return mag;
189 }
190 case modifier::heaviside:
191 {
192 double psi0 = jsmod.get( "boundary", 1.1 ).asDouble();
193 double alpha = jsmod.get( "alpha", 0.2 ).asDouble();
194 double sign = +1;
195 if( desc == description::standardX || desc == description::standardO ||
196 desc == description::doubleX)
197 detail::transform_psi( mag, psi0, alpha, sign);
198 else
199 sign = jsmod.get( "sign", -1. ).asDouble();
200
201 mod_psip = mod::createPsip( mod::everywhere, mag.get_psip(), psi0, alpha, sign);
202 wall = mod::DampingRegion( mod::everywhere, mag.psip(), psi0, alpha, -sign);
203 transition = mod::MagneticTransition( mod::everywhere, mag.psip(), psi0, alpha, sign);
204 break;
205 }
206 case modifier::sol_pfr:
207 {
208 double psi0 = jsmod["boundary"].get( 0, 1.1 ).asDouble();
209 double alpha0 = jsmod["alpha"].get(0, 0.2 ).asDouble();
210 double psi1 = jsmod["boundary"].get(1, 0.97 ).asDouble();
211 double alpha1 = jsmod[ "alpha"].get(1, 0.2 ).asDouble();
212 switch( desc){
213 case description::standardX:
214 {
215 double sign0 = +1., sign1 = -1.;
216 detail::transform_psi( mag, psi0, alpha0, sign0);
217 detail::transform_psi( mag, psi1, alpha1, sign1);
218 //we can find the X-point
219 double RX = mag.R0()-1.1*mag.params().triangularity()*mag.params().a();
220 double ZX = -1.1*mag.params().elongation()*mag.params().a();
221 dg::geo::findXpoint( mag.get_psip(), RX, ZX);
222 CylindricalFunctorsLvl2 mod0_psip;
223 mod0_psip = mod::createPsip(
224 mod::everywhere, mag.get_psip(), psi0, alpha0, sign0);
225 mod_psip = mod::createPsip(
226 mod::HeavisideZ( ZX, -1), mod0_psip, psi1, alpha1, sign1);
227 CylindricalFunctor wall0 = mod::DampingRegion( mod::everywhere, mag.psip(), psi0, alpha0, -sign0);
228 CylindricalFunctor transition0 = mod::MagneticTransition( mod::everywhere, mag.psip(), psi0, alpha0, sign0);
229 CylindricalFunctor wall1 = mod::DampingRegion( mod::HeavisideZ(ZX, -1), mag.psip(), psi1, alpha1, -sign1);
230 CylindricalFunctor transition1 = mod::MagneticTransition( mod::HeavisideZ(ZX, -1), mag.psip(), psi1, alpha1, sign1);
231 wall = mod::SetUnion( wall0, wall1);
232 transition = mod::SetUnion( transition0, transition1);
233 break;
234 }
235 case description::doubleX:
236 {
237 double sign0 = +1., sign1 = -1.;
238 detail::transform_psi( mag, psi0, alpha0, sign0);
239 detail::transform_psi( mag, psi1, alpha1, sign1);
240 //we can find the X-point
241 double RX1 = mag.R0()-1.1*mag.params().triangularity()*mag.params().a();
242 double ZX1 = -1.1*mag.params().elongation()*mag.params().a();
243 dg::geo::findXpoint( mag.get_psip(), RX1, ZX1);
244 double RX2 = mag.R0()-1.1*mag.params().triangularity()*mag.params().a();
245 double ZX2 = +1.1*mag.params().elongation()*mag.params().a();
246 dg::geo::findXpoint( mag.get_psip(), RX2, ZX2);
247 CylindricalFunctorsLvl2 mod0_psip, mod1_psip;
248 mod0_psip = mod::createPsip(
249 mod::everywhere, mag.get_psip(), psi0, alpha0, sign0);
250 mod1_psip = mod::createPsip(
251 mod::HeavisideZ( ZX1, -1), mod0_psip, psi1, alpha1, sign1);
252 mod_psip = mod::createPsip(
253 mod::HeavisideZ( ZX2, +1), mod1_psip, psi1, alpha1, sign1);
254 CylindricalFunctor wall0 = mod::DampingRegion( mod::everywhere, mag.psip(), psi0, alpha0, -sign0);
255 CylindricalFunctor wall1 = mod::DampingRegion( mod::HeavisideZ(ZX1, -1), mag.psip(), psi1, alpha1, -sign1);
256 CylindricalFunctor wall2 = mod::DampingRegion( mod::HeavisideZ(ZX2, +1), mag.psip(), psi1, alpha1, -sign1);
257 CylindricalFunctor transition0 = mod::MagneticTransition( mod::everywhere, mag.psip(), psi0, alpha0, sign0);
258 CylindricalFunctor transition1 = mod::MagneticTransition( mod::HeavisideZ(ZX1, -1), mag.psip(), psi1, alpha1, sign1);
259 CylindricalFunctor transition2 = mod::MagneticTransition( mod::HeavisideZ(ZX2, +1), mag.psip(), psi1, alpha1, sign1);
260 transition = mod::SetUnion( mod::SetUnion( transition0, transition1), transition2);
261 wall = mod::SetUnion( mod::SetUnion( wall0, wall1), wall2);
262 break;
263 }
264 default:
265 {
266 double sign0 = jsmod[ "sign"].get( 0, -1. ).asDouble();
267 double sign1 = jsmod[ "sign"].get( 1, +1. ).asDouble();
268 CylindricalFunctorsLvl2 mod0_psip;
269 mod0_psip = mod::createPsip(
270 mod::everywhere, mag.get_psip(), psi0, alpha0, sign0);
271 mod_psip = mod::createPsip(
272 mod::everywhere, mod0_psip, psi1, alpha1, sign1);
273 CylindricalFunctor wall0 = mod::DampingRegion( mod::everywhere, mag.psip(), psi0, alpha0, sign0);
274 CylindricalFunctor transition0 = mod::MagneticTransition( mod::everywhere, mag.psip(), psi0, alpha0, sign0);
275 CylindricalFunctor wall1 = mod::DampingRegion( mod::everywhere, mag.psip(), psi1, alpha1, sign1);
276 CylindricalFunctor transition1 = mod::MagneticTransition( mod::everywhere, mag.psip(), psi1, alpha1, sign1);
277 wall = mod::SetUnion( wall0, wall1);
278 transition = mod::SetUnion( transition0, transition1);
279 break;
280 }
281 }
282 }
283 }
284 switch( equi){
285 case equilibrium::solovev:
286 {
287 solovev::Parameters gp( gs);
288 return TokamakMagneticField( gp.R_0,mod_psip,
289 solovev::createIpol( gp, mod_psip), mod_params);
290 }
291 default:
292 {
293 return TokamakMagneticField( mag.R0(), mod_psip,
294 mag.get_ipol(), mod_params);
295 }
296 }
297}

337static inline void createSheathRegion(
338 dg::file::WrappedJsonValue jsmod, TokamakMagneticField mag,
339 CylindricalFunctor wall, dg::Grid2d& sheath_walls,
340 CylindricalFunctor& sheath)
341{
342 double R0 = sheath_walls.x0(), R1 = sheath_walls.x1();
343 double Z0 = sheath_walls.y0(), Z1 = sheath_walls.y1();
344 Grid1d gR1d ( R0, R1, 1, 100);
345 Grid1d gZ1d ( Z0, Z1, 1, 100);
346 std::array<bool,2> sheathR = {false,false}, sheathZ = {false,false};
347 for ( unsigned i=0; i<100; i++)
348 {
349 if( wall( R0+i*gR1d.h(), Z0) == 0)
350 sheathR[0] = true;
351 if( wall( R0+i*gR1d.h(), Z1) == 0)
352 sheathR[1] = true;
353 if( wall( R0, Z0 + i*gZ1d.h()) == 0)
354 sheathZ[0] = true;
355 if( wall( R1, Z0 + i*gZ1d.h()) == 0)
356 sheathZ[1] = true;
357 }
358 if( false == sheathR[0]) Z0 = -1e10;
359 if( false == sheathR[1]) Z1 = 1e10;
360 if( false == sheathZ[0]) R0 = -1e10;
361 if( false == sheathZ[1]) R1 = 1e10;
362 sheath_walls = dg::Grid2d( R0, R1, Z0, Z1, 1,1,1);
363 double boundary = jsmod.get( "boundary", 0.0625 ).asDouble(); // 1/16
364 double alpha = jsmod.get( "alpha", 0.015625 ).asDouble(); // 1/64
365 CylindricalFunctor distM = dg::geo::WallFieldlineDistance( dg::geo::createBHat(
366 mag), sheath_walls, (-boundary-1e-3)*2.0*M_PI,
367 1e-6, "phi");
368 CylindricalFunctor distP = dg::geo::WallFieldlineDistance( dg::geo::createBHat(
369 mag), sheath_walls, (+boundary+1e-3)*2.0*M_PI,
370 1e-6, "phi");
371 dg::PolynomialHeaviside polyM( -boundary*2.*M_PI + alpha*M_PI, alpha*M_PI, +1);
372 dg::PolynomialHeaviside polyP( boundary*2.*M_PI - alpha*M_PI, alpha*M_PI, -1);
373 auto sheathM = dg::compose( polyM, distM); //positive (because distance)
374 auto sheathP = dg::compose( polyP, distP);
375 sheath = mod::SetUnion( sheathM, sheathP);
376 sheath = mod::SetIntersection( mod::SetNot( wall), sheath);
377}
