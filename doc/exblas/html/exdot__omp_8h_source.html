70  std::vector<int64_t>& acc, int const linesize)

72  // Custom reduction
73  for(unsigned int s = 1; (unsigned)(1 << (s-1)) < tnum; ++s)
74  {
75  int32_t volatile * c = &ready[tid * linesize];
76  ++*c;
77  if(tid % (1 << s) == 0) {
78  unsigned int tid2 = tid | (1 << (s-1));
