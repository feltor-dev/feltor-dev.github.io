42 inline static void ReductionStep(int step, int64_t * acc1, int64_t * acc2,
43  int volatile * ready)
44 {
45 #ifndef _WITHOUT_VCL
46  _mm_prefetch((char const*)ready, _MM_HINT_T0);
47  // Wait for thread 2 to be ready
48  while(*ready < step) {
49  // wait
50  _mm_pause();
51  }
52 #endif//_WITHOUT_VCL
53  int imin = IMIN, imax = IMAX;
54  Normalize( acc1, imin, imax);
55  imin = IMIN, imax = IMAX;
56  Normalize( acc2, imin, imax);
57  for(int i = IMIN; i <= IMAX; ++i) {
58  acc1[i] += acc2[i];
59  }
60 }

69 inline static void Reduction(unsigned int tid, unsigned int tnum, std::vector<int32_t>& ready,
70  std::vector<int64_t>& acc, int const linesize)
71 {
72  // Custom tree reduction
73  for(unsigned int s = 1; (unsigned)(1 << (s-1)) < tnum; ++s)
74  {
75  // 1<<(s-1) = 0001, 0010, 0100, ... = 1,2,4,8,16,...
76  int32_t volatile * c = &ready[tid * linesize];
77  ++*c; //set: ready for level s
78 #ifdef _WITHOUT_VCL
79 #pragma omp barrier //all threads are ready for level s
80 #endif
81  if(tid % (1 << s) == 0) { //1<<s = 2,4,8,16,32,...
82  //only the tid thread executes this block, tid2 just sets ready
83  unsigned int tid2 = tid | (1 << (s-1)); //effectively adds 1, 2, 4,...
84  if(tid2 < tnum) {
85  ReductionStep(s, &acc[tid*BIN_COUNT], &acc[tid2*BIN_COUNT],
86  &ready[tid2 * linesize]);
87  }
88  }
89  }
90 }
