74 static inline void AccumulateWord( int64_t *accumulator, int i, int64_t x) {
75  // With atomic accumulator updates
76  // accumulation and carry propagation can happen in any order,
77  // as long as addition is atomic
78  // only constraint is: never forget an overflow bit
79  //assert(i >= 0 && i < BIN_COUNT);
80  unsigned char overflow;
81  int64_t carry = x;
82  int64_t carrybit;
83  int64_t oldword = cpu::xadd(accumulator[i], x, overflow);
84  // To propagate over- or underflow
85  while(unlikely(overflow)) {
86  // Carry or borrow
87  // oldword has sign S
88  // x has sign S
89  // accumulator[i] has sign !S (just after update)
90  // carry has sign !S
91  // carrybit has sign S
92  carry = (oldword + carry) >> DIGITS; // Arithmetic shift
93  bool s = oldword > 0;
94  carrybit = (s ? 1ll << KRX : -1ll << KRX);
95 
96  // Cancel carry-save bits
97  cpu::xadd(accumulator[i], (int64_t) -(carry << DIGITS), overflow);
98  if(TSAFE && unlikely(s ^ overflow)) {
99  // (Another) overflow of sign S
100  carrybit *= 2;
101  }
102  carry += carrybit;
103 
104  ++i;
105  if (i >= BIN_COUNT){
106  //status = Overflow;
107  return;
108  }
109  oldword = cpu::xadd(accumulator[i], carry, overflow);
110  }
111 }

121 static inline void Accumulate( int64_t* accumulator, double x) {
122  if (x == 0)
123  return;
124 
125 
126  int e = cpu::exponent(x);
127  int exp_word = e / DIGITS; // Word containing MSbit (upper bound)
128  int iup = exp_word + F_WORDS;
129 
130  double xscaled = cpu::myldexp(x, -DIGITS * exp_word);
131 
132  int i;
133  for (i = iup; xscaled != 0; --i) {
134  double xrounded = cpu::myrint(xscaled);
135  int64_t xint = cpu::myllrint(xscaled);
136  AccumulateWord(accumulator, i, xint);
137 
138  xscaled -= xrounded;
139  xscaled *= DELTASCALE;
140  }

171 static inline bool Normalize( int64_t *accumulator, int& imin, int& imax) {
172  int64_t carry_in = accumulator[imin] >> DIGITS;
173  accumulator[imin] -= carry_in << DIGITS;
174  int i;
175  // Sign-extend all the way
176  for (i = imin + 1; i < BIN_COUNT; ++i) {
177  accumulator[i] += carry_in;
178  int64_t carry_out = accumulator[i] >> DIGITS; // Arithmetic shift
179  accumulator[i] -= (carry_out << DIGITS);
180  carry_in = carry_out;
181  }
182  imax = i - 1;
183 
184  // Do not cancel the last carry to avoid losing information
185  accumulator[imax] += carry_in << DIGITS;
186 
187  return carry_in < 0;
188 }

202 static inline double Round( int64_t * accumulator) {
203  int imin = IMIN;
204  int imax = IMAX;
205  bool negative = Normalize(accumulator, imin, imax);
206 
207  // Find leading word
208  int i;
209  // Skip zeroes
210  for(i = imax; accumulator[i] == 0 && i >= imin; --i) {
211  }
212  if (negative) {
213  // Skip ones
214  for(; (accumulator[i] & ((1ll << DIGITS) - 1)) == ((1ll << DIGITS) - 1) && i >= imin; --i) {
215  }
216  }
217  if (i < 0) {
218  return 0.0;
219  }
220 
221  int64_t hiword = negative ? ((1ll << DIGITS) - 1) - accumulator[i] : accumulator[i];
222  double rounded = (double)hiword;
223  double hi = ldexp(rounded, (i - F_WORDS) * DIGITS);
224  if (i == 0) {
225  return negative ? -hi : hi; // Correct rounding achieved
226  }
227  hiword -= std::llrint(rounded);
228  double mid = ldexp((double) hiword, (i - F_WORDS) * DIGITS);
229 
230  // Compute sticky
231  int64_t sticky = 0;
232  for (int j = imin; j != i - 1; ++j) {
233  sticky |= negative ? ((1ll << DIGITS) - accumulator[j]) : accumulator[j];
234  }
235 
236  int64_t loword = negative ? ((1ll << DIGITS) - accumulator[i - 1]) : accumulator[i - 1];
237  loword |= !!sticky;
238  double lo = ldexp((double) loword, (i - 1 - F_WORDS) * DIGITS);
239 
240  // Now add3(hi, mid, lo)
241  // No overlap, we have already normalized
242  if (mid != 0) {
243  lo = cpu::OddRoundSumNonnegative(mid, lo);
244  }
245  // Final rounding
246  hi = hi + lo;
247  return negative ? -hi : hi;
248 }
