228  {
229  if( bcx == dg::PER)
230  {
231  real_type N = floor((x-x0_)/(x1_-x0_)); // ... -2[ -1[ 0[ 1[ 2[ ...
232  x = x - N*(x1_-x0_); //shift
233  }
234  //mirror along boundary as often as necessary
235  while( (x<x0_) || (x>x1_) )
236  {
237  if( x < x0_){
238  x = 2.*x0_ - x;
239  //every mirror swaps the sign if Dirichlet
240  if( bcx == dg::DIR || bcx == dg::DIR_NEU)
241  negative = !negative;//swap sign
242  }
243  if( x > x1_){
244  x = 2.*x1_ - x;
245  if( bcx == dg::DIR || bcx == dg::NEU_DIR) //notice the different boundary NEU_DIR to the above DIR_NEU !
246  negative = !negative; //swap sign
247  }
