220 void shift( bool& negative, real_type &x, bc bcx)const
221 {
222 if( bcx == dg::PER)
223 {
224 real_type N = floor((x-x0_)/(x1_-x0_)); // ... -2[ -1[ 0[ 1[ 2[ ...
225 x = x - N*(x1_-x0_); //shift
226 }
227 //mirror along boundary as often as necessary
228 while( (x<x0_) || (x>x1_) )
229 {
230 if( x < x0_){
231 x = 2.*x0_ - x;
232 //every mirror swaps the sign if Dirichlet
233 if( bcx == dg::DIR || bcx == dg::DIR_NEU)
234 negative = !negative;//swap sign
235 }
236 if( x > x1_){
237 x = 2.*x1_ - x;
238 if( bcx == dg::DIR || bcx == dg::NEU_DIR) //notice the different boundary NEU_DIR to the above DIR_NEU !
239 negative = !negative; //swap sign
240 }
241 }
