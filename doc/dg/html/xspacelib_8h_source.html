67template< class T>
68cusp::coo_matrix< int, T, cusp::host_memory> tensorproduct(
69 const cusp::coo_matrix< int, T, cusp::host_memory>& lhs,
70 const cusp::coo_matrix< int, T, cusp::host_memory>& rhs)
71{
72 //assert quadratic matrices
73 assert( lhs.num_rows == lhs.num_cols);
74 assert( rhs.num_rows == rhs.num_cols);
75 //assert dg matrices
76 unsigned Nx = rhs.num_rows;
77 //taken from the cusp examples:
78 //dimensions of the matrix
79 int num_rows = lhs.num_rows*rhs.num_rows;
80 int num_cols = num_rows;
81 int num_triplets = lhs.num_entries*rhs.num_entries;
82 // allocate storage for unordered triplets
83 cusp::array1d< int, cusp::host_memory> I( num_triplets); // row indices
84 cusp::array1d< int, cusp::host_memory> J( num_triplets); // column indices
85 cusp::array1d< T, cusp::host_memory> V( num_triplets); // values
86 //fill triplet arrays
87 detail::AddIndex2d addIndexRow( Nx);
88 detail::AddIndex2d addIndexCol( Nx);
89 detail::AddIndex2d addIndexVal( Nx);
90 //LHS x RHS
91 for( unsigned i=0; i<lhs.num_entries; i++)
92 for( unsigned j=0; j<rhs.num_entries; j++)
93 {
94 addIndexRow( I, lhs.row_indices[i], rhs.row_indices[j]);
95 addIndexCol( J, lhs.column_indices[i], rhs.column_indices[j]);
96 addIndexVal( V, lhs.values[i]*rhs.values[j]);
97 }
98 cusp::array1d< int, cusp::host_memory> dI( I); // row indices
99 cusp::array1d< int, cusp::host_memory> dJ( J); // column indices
100 cusp::array1d< T, cusp::host_memory> dV( V); // values
101 // sort triplets by (i,j) index using two stable sorts (first by J, then by I)
102 thrust::stable_sort_by_key(dJ.begin(), dJ.end(), thrust::make_zip_iterator(thrust::make_tuple(dI.begin(), dV.begin())));
103 thrust::stable_sort_by_key(dI.begin(), dI.end(), thrust::make_zip_iterator(thrust::make_tuple(dJ.begin(), dV.begin())));
104 // compute unique number of ( values with different (i,j) index) in the output
105 int num_entries = thrust::inner_product(thrust::make_zip_iterator(thrust::make_tuple(dI.begin(), dJ.begin())),
106 thrust::make_zip_iterator(thrust::make_tuple(dI.end (), dJ.end())) - 1,
107 thrust::make_zip_iterator(thrust::make_tuple(dI.begin(), dJ.begin())) + 1,
108 int(0),
109 thrust::plus<int>(),
110 thrust::not_equal_to< thrust::tuple<int,int> >()) + 1;
111
112 // allocate output matrix
113 cusp::coo_matrix<int, T, cusp::host_memory> A(num_rows, num_cols, num_entries);
114 // sum values with the same (i,j) index
115 thrust::reduce_by_key(thrust::make_zip_iterator(thrust::make_tuple(dI.begin(), dJ.begin())),
116 thrust::make_zip_iterator(thrust::make_tuple(dI.end(), dJ.end())),
117 dV.begin(),
118 thrust::make_zip_iterator(thrust::make_tuple(A.row_indices.begin(), A.column_indices.begin())),
119 A.values.begin(),
120 thrust::equal_to< thrust::tuple<int,int> >(),
121 thrust::plus<T>());
122 return A;
123}

138template<class real_type>
139dg::IHMatrix backscatter( const aRealTopology2d<real_type>& g)
140{
141 typedef cusp::coo_matrix<int, real_type, cusp::host_memory> Matrix;
142 //create equidistant backward transformation
143 dg::Operator<real_type> backwardeqX( g.dltx().backwardEQ());
144 dg::Operator<real_type> backwardeqY( g.dlty().backwardEQ());
145 dg::Operator<real_type> forwardX( g.dltx().forward());
146 dg::Operator<real_type> forwardY( g.dlty().forward());
147 dg::Operator<real_type> backward1dX = backwardeqX*forwardX;
148 dg::Operator<real_type> backward1dY = backwardeqY*forwardY;
149
150 Matrix transformX = dg::tensorproduct( g.Nx(), backward1dX);
151 Matrix transformY = dg::tensorproduct( g.Ny(), backward1dY);
152 Matrix backward = dg::tensorproduct( transformY, transformX);
153
154 return (dg::IHMatrix)backward;
155
156}
