372 T lr_pivot( dg::Operator<T>& m, std::vector<unsigned>& p)
373 {
374  //from numerical recipes
375  T pivot, determinant=(T)1;
376  unsigned pivotzeile, numberOfSwaps=0;
377  const size_t n = m.size();
378  p.resize( n);
379  for( size_t j = 0; j < n; j++) //gehe Spalten /Diagonale durch
380  {
381  //compute upper matrix except for the diagonal element (the pivot)
382  for( size_t i = 0; i< j; i++)
383  for( size_t k=0; k<i; k++)
384  m(i,j)-=m(i,k)*m(k,j);
385  //compute canditates for pivot elements
386  for( size_t i = j; i< n; i++)
387  for( size_t k=0; k<j; k++)
388  m(i,j)-=m(i,k)*m(k,j);
389  //search for absolute maximum of pivot candidates
390  pivot = m(j,j);
391  pivotzeile = j;
392  for( size_t i = j+1; i < n; i++)
393  if( fabs( m(i,j)) > fabs(pivot))
394  {
395  pivot = m(i,j), pivotzeile = i;
396  }
397 
398  if( pivot!= (T)0 )
399  {
400  if( pivotzeile != j)
401  {
402  m.swap_lines( pivotzeile, j);
403  numberOfSwaps++;
404  }
405  p[j] = pivotzeile;
406  //divide all elements below the diagonal by the pivot to get the lower matrix
407  for( size_t i=j+1; i<n; i++)
408  m(i,j) /= pivot;
409  determinant*=m(j,j);
410 
411  }
412  else
413  throw std::runtime_error( "Matrix is singular!!");
414  }
415  if( numberOfSwaps % 2 != 0)
416  determinant*=-1.;
417  return determinant;
