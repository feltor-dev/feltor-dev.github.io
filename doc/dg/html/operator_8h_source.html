376 T lr_pivot( dg::Operator<T>& m, std::vector<unsigned>& p)
377 {
378  //from numerical recipes
379  T pivot, determinant=(T)1;
380  unsigned pivotzeile, numberOfSwaps=0;
381  const size_t n = m.size();
382  p.resize( n);
383  for( size_t j = 0; j < n; j++) //gehe Spalten /Diagonale durch
384  {
385  //compute upper matrix except for the diagonal element (the pivot)
386  for( size_t i = 0; i< j; i++)
387  for( size_t k=0; k<i; k++)
388  m(i,j)-=m(i,k)*m(k,j);
389  //compute canditates for pivot elements
390  for( size_t i = j; i< n; i++)
391  for( size_t k=0; k<j; k++)
392  m(i,j)-=m(i,k)*m(k,j);
393  //search for absolute maximum of pivot candidates
394  pivot = m(j,j);
395  pivotzeile = j;
396  for( size_t i = j+1; i < n; i++)
397  if( fabs( m(i,j)) > fabs(pivot))
398  {
399  pivot = m(i,j), pivotzeile = i;
400  }
401 
402  if( pivot!= (T)0 )
403  {
404  if( pivotzeile != j)
405  {
406  m.swap_lines( pivotzeile, j);
407  numberOfSwaps++;
408  }
409  p[j] = pivotzeile;
410  //divide all elements below the diagonal by the pivot to get the lower matrix
411  for( size_t i=j+1; i<n; i++)
412  m(i,j) /= pivot;
413  determinant*=m(j,j);
414 
415  }
416  else
417  throw std::runtime_error( "Matrix is singular!!");
418  }
419  if( numberOfSwaps % 2 != 0)
420  determinant*=-1.;
421  return determinant;
