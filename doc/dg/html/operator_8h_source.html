373template< class T>
374T lu_pivot( dg::Operator<T>& m, std::vector<unsigned>& p)
375{
376 //from numerical recipes
377 T pivot, determinant=(T)1;
378 unsigned pivotzeile, numberOfSwaps=0;
379 const size_t n = m.size();
380 p.resize( n);
381 for( size_t j = 0; j < n; j++) //gehe Spalten /Diagonale durch
382 {
383 //compute upper matrix except for the diagonal element (the pivot)
384 for( size_t i = 0; i< j; i++)
385 {
386 thrust::host_vector<T> mik(i), mkj(i);
387 for( size_t k=0; k<i; k++)
388 mik[k] = m(i,k), mkj[k] = m(k,j);
389 m(i,j) -= dg::blas1::dot( mik, mkj);
390 }
391 //compute canditates for pivot elements
392 for( size_t i = j; i< n; i++)
393 {
394 thrust::host_vector<T> mik(j), mkj(j);
395 for( size_t k=0; k<j; k++)
396 mik[k] = m(i,k), mkj[k] = m(k,j);
397 m(i,j) -= dg::blas1::dot( mik, mkj);
398 }
399 //search for absolute maximum of pivot candidates
400 pivot = m(j,j);
401 pivotzeile = j;
402 for( size_t i = j+1; i < n; i++)
403 if( fabs( m(i,j)) > fabs(pivot))
404 {
405 pivot = m(i,j), pivotzeile = i;
406 }
407
408 if( fabs(pivot) > 1e-15 )
409 {
410 if( pivotzeile != j)
411 {
412 m.swap_lines( pivotzeile, j);
413 numberOfSwaps++;
414 }
415 p[j] = pivotzeile;
416 //divide all elements below the diagonal by the pivot to get the lower matrix
417 for( size_t i=j+1; i<n; i++)
418 m(i,j) /= pivot;
419 determinant*=m(j,j);
420
421 }
422 else
423 throw std::runtime_error( "Matrix is singular!!");
424 }
425 if( numberOfSwaps % 2 != 0)
426 determinant*=-1.;
427 return determinant;
428

440template<class T>
441void lu_solve( const dg::Operator<T>& lu, const std::vector<unsigned>& p, std::vector<T>& b)
442{
443 assert(p.size() == lu.size() && p.size() == b.size());
444 const size_t n = p.size();
445 // Vorwrtseinsetzen
446 for( size_t i = 0; i<n; i++)
447 {
448 //mache Zeilentausch
449 std::swap( b[ p[i] ], b[i]);
450 thrust::host_vector<T> lui(i), bi(i);
451 for( size_t j = 0; j < i; j++)
452 lui[j] = lu(i,j), bi[j] = b[j];
453 b[i] -= dg::blas1::dot( lui, bi);
454 }
455 // Rckwrtseinsetzen
456 for( int i = n-1; i>=0; i--)
457 {
458 thrust::host_vector<T> lui(n-(i+1)), bi(n-(i+1));
459 for( size_t j = i+1; j < n; j++)
460 lui[j-(i+1)] = lu(i,j), bi[j-(i+1)] = b[j];
461 b[i] -= dg::blas1::dot( lui, bi);
462 b[i] /= lu(i,i);
463 }

479dg::Operator<T> inverse( const dg::Operator<T>& in)
480{
481 dg::Operator<T> out(in);
482 const unsigned n = in.size();
483 std::vector<unsigned> pivot( n);
484 dg::Operator<T> lu(in);
485 T determinant = lu_pivot( lu, pivot);
486 if( fabs(determinant ) < 1e-14)
487 throw std::runtime_error( "Determinant zero!");
488 for( unsigned i=0; i<n; i++)
489 {
490 std::vector<T> unit(n, 0);
491 unit[i] = 1;
492 lu_solve( lu, pivot, unit);
493 for( unsigned j=0; j<n; j++)
494 out(j,i) = unit[j];
495 }
