255void ImExMultistep<ContainerType>::step( const std::tuple<RHS, Diffusion, Solver>& ode, value_type& t, ContainerType& u)
256{
257 unsigned s = m_t.steps();
258 if( m_counter < s - 1)
259 {
260 std::map<unsigned, std::string> order2method{
261 {1, "ARK-4-2-3"},
262 {2, "ARK-4-2-3"},
263 {3, "ARK-4-2-3"},
264 {4, "ARK-6-3-4"},
265 {5, "ARK-8-4-5"},
266 {6, "ARK-8-4-5"},
267 {7, "ARK-8-4-5"}
268 };
269 ARKStep<ContainerType> ark( order2method.at( m_t.order()), u);
270 ContainerType tmp ( u);
271 ark.step( ode, t, u, t, u, m_dt, tmp);
272 m_counter++;
273 m_tu = t;
274 dg::blas1::copy( u, m_u[s-1-m_counter]);
275 if( s-1-m_counter < m_im.size())
276 std::get<1>(ode)( m_tu, m_u[s-1-m_counter], m_im[s-1-m_counter]);
277 std::get<0>(ode)( m_tu, m_u[s-1-m_counter], m_ex[s-1-m_counter]);
278 return;
279 }
280 //compute right hand side of inversion equation
281 dg::blas1::axpbypgz( m_t.a(0), m_u[0], m_dt*m_t.ex(0), m_ex[0], 0., m_tmp);
282 for (unsigned i = 1; i < s; i++)
283 dg::blas1::axpbypgz( m_t.a(i), m_u[i], m_dt*m_t.ex(i), m_ex[i], 1., m_tmp);
284 for (unsigned i = 0; i < m_im.size(); i++)
285 dg::blas1::axpby( m_dt*m_t.im(i+1), m_im[i], 1., m_tmp);
286 t = m_tu = m_tu + m_dt;
287
288 //Rotate 1 to the right (note the reverse iterator here!)
289 std::rotate( m_u.rbegin(), m_u.rbegin() + 1, m_u.rend());
290 std::rotate( m_ex.rbegin(), m_ex.rbegin() + 1, m_ex.rend());
291 if( !m_im.empty())
292 std::rotate( m_im.rbegin(), m_im.rbegin() + 1, m_im.rend());
293 //compute implicit part
294 value_type alpha = m_dt*m_t.im(0);
295 std::get<2>(ode)( alpha, t, u, m_tmp);
296
297 blas1::copy( u, m_u[0]); //store result
298 if( 0 < m_im.size())
299 dg::blas1::axpby( 1./alpha, u, -1./alpha, m_tmp, m_im[0]);
300 std::get<0>(ode)(m_tu, m_u[0], m_ex[0]); //call f on new point (AFTER diff!)
301}

721 value_type t_begin, const ContainerType&
722 begin, value_type& t_end, ContainerType& end,
723 bool check ) const
724{
725 bool forward = (t_end - t_begin > 0);
726 if( (m_dt < 0 && forward) || ( m_dt > 0 && !forward) )
