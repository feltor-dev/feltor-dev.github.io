191 {
192  unsigned s = m_t.steps();
193  if( m_counter < s - 1)
194  {
195  std::map<unsigned, std::string> order2method{
196  {1, "ARK-4-2-3"},
197  {2, "ARK-4-2-3"},
198  {3, "ARK-4-2-3"},
199  {4, "ARK-6-3-4"},
200  {5, "ARK-8-4-5"},
201  {6, "ARK-8-4-5"},
202  {7, "ARK-8-4-5"}
203  };
204  ARKStep<ContainerType, SolverType> ark( order2method.at( m_t.order()), m_solver);
205  ContainerType tmp ( u);
206  ark.step( f, diff, t, u, t, u, m_dt, tmp);
207  m_counter++;
208  m_tu = t;
209  dg::blas1::copy( u, m_u[s-1-m_counter]);
210  f( m_tu, m_u[s-1-m_counter], m_ex[s-1-m_counter]);
211  //only assign to f if we actually need to store it
212  if( s-1-m_counter < m_im.size())
213  diff( m_tu, m_u[s-1-m_counter], m_im[s-1-m_counter]);
214  m_solver = ark.solver(); // store the state of the solver
215  return;
216  }
217  //compute right hand side of inversion equation
218  dg::blas1::axpbypgz( m_t.a(0), m_u[0], m_dt*m_t.ex(0), m_ex[0], 0., m_tmp);
219  for (unsigned i = 1; i < s; i++)
220  dg::blas1::axpbypgz( m_t.a(i), m_u[i], m_dt*m_t.ex(i), m_ex[i], 1., m_tmp);
221  for (unsigned i = 0; i < m_im.size(); i++)
222  dg::blas1::axpby( m_dt*m_t.im(i+1), m_im[i], 1., m_tmp);
223  t = m_tu = m_tu + m_dt;
224 
225  value_type alpha[2] = {2., -1.};
226  //value_type alpha[2] = {1., 0.};
227  if( s > 1 ) //everything higher than Euler
228  dg::blas1::axpby( alpha[0], m_u[0], alpha[1], m_u[1], u);
229  else
230  dg::blas1::copy( m_u[0], u);
231 
232  //Rotate 1 to the right (note the reverse iterator here!)
233  std::rotate( m_u.rbegin(), m_u.rbegin() + 1, m_u.rend());
234  std::rotate( m_ex.rbegin(), m_ex.rbegin() + 1, m_ex.rend());
235  if( !m_im.empty())
236  std::rotate( m_im.rbegin(), m_im.rbegin() + 1, m_im.rend());
237  //compute implicit part
238  m_solver.solve( -m_dt*m_t.im(0), diff, t, u, m_tmp);
239 
240  blas1::copy( u, m_u[0]); //store result
241  f(m_tu, m_u[0], m_ex[0]); //call f on new point
242  if( 0 < m_im.size())
243  diff( m_tu, m_u[0], m_im[0]); //call diff on new point
