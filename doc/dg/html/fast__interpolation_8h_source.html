69 int dims = inter_.size();

72 dg::blas2::symv( alpha, inter_[0], x, beta, y);

78 dg::blas2::symv( alpha, inter_[dims-1], temp_[dims-2].data(), beta, y);
79 }

121template<class real_type>
122MultiMatrix< dg::HMatrix_t<real_type>, dg::HVec_t<real_type> > fast_interpolation( const RealGrid1d<real_type>& t, unsigned multiplyn, unsigned multiplyNx)
123{
124 unsigned n=t.n();
125 dg::RealGrid1d<real_type> g_old( -1., 1., n, 1);
126 dg::RealGrid1d<real_type> g_new( -1., 1., n*multiplyn, multiplyNx);
127 cusp::coo_matrix<int, real_type, cusp::host_memory> interpolX = dg::create::interpolation( g_new, g_old);
128 EllSparseBlockMat<real_type> iX( multiplyn*multiplyNx*t.N(), t.N(), 1, multiplyNx*multiplyn, t.n());
129 for( unsigned k=0; k<multiplyNx*multiplyn; k++)
130 for( unsigned i=0; i<n; i++)
131 for( unsigned j=0; j<n; j++)
132 iX.data[(k*n+i)*n+j] = interpolX.values[(k*n+i)*n+j];
133 for( unsigned i=0; i<multiplyNx*multiplyn*t.N(); i++)
134 {
135 iX.cols_idx[i] = i/(multiplyNx*multiplyn);
136 iX.data_idx[i] = i%(multiplyNx*multiplyn);
137 }
138 MultiMatrix < EllSparseBlockMat<real_type>, thrust::host_vector<real_type> > inter(1);
139 inter.get_matrices()[0] = iX;
140 return inter;
141}

160template<class real_type>
161MultiMatrix< dg::HMatrix_t<real_type>, dg::HVec_t<real_type> > fast_projection( const RealGrid1d<real_type>& t, unsigned dividen, unsigned divideNx)
162{
163 if( t.N()%divideNx != 0) throw Error( Message(_ping_)<< "Nx and divideNx don't match: Nx: " << t.N()<< " divideNx "<< (unsigned)divideNx);
164 if( t.n()%dividen != 0) throw Error( Message(_ping_)<< "n and dividen don't match: n: " << t.n()<< " dividen "<< (unsigned)dividen);
165 unsigned n=t.n()/dividen;
166 dg::RealGrid1d<real_type> g_old( -1., 1., n*dividen, divideNx);
167 dg::RealGrid1d<real_type> g_new( -1., 1., n, 1);
168 dg::HVec w1d = dg::create::weights( g_old);
169 dg::HVec v1d = dg::create::inv_weights( g_new);
170 cusp::coo_matrix<int, real_type, cusp::host_memory> projectX;
171 //Here, we cannot use create::projection because that would remove explicit zeros!!
172 projectX = dg::create::interpolationT( g_new, g_old);
173 EllSparseBlockMat<real_type> pX( t.N()/divideNx, t.N()*dividen, divideNx*dividen, divideNx*dividen, n);
174 for( unsigned k=0; k<divideNx; k++)
175 for( unsigned l=0; l<dividen; l++)
176 for( unsigned i=0; i<n; i++)
177 for( unsigned j=0; j<n; j++)
178 {
179 pX.data[((k*dividen+l)*n+i)*n+j] = projectX.values[((i*divideNx+k)*dividen + l)*n+j];
180 pX.data[((k*dividen+l)*n+i)*n+j] *= v1d[i]*w1d[l*n+j];
181 }
182 for( unsigned i=0; i<t.N()/divideNx; i++)
183 for( unsigned d=0; d<divideNx*dividen; d++)
184 {
185 pX.cols_idx[i*divideNx*dividen+d] = i*divideNx*dividen+d;
186 pX.data_idx[i*divideNx*dividen+d] = d;
187 }
188 MultiMatrix < EllSparseBlockMat<real_type>, thrust::host_vector<real_type> > inter(1);
189 inter.get_matrices()[0] = pX;
190 return inter;
191}

197{

208 inter.get_matrices()[0] = interX.get_matrices()[0];
209 inter.get_matrices()[1] = interY.get_matrices()[0];
210 thrust::host_vector<real_type> vec( t.size()*multiplyNx*multiplyn);
