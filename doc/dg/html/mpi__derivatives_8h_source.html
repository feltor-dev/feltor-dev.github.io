28 CooSparseBlockMat<double> save_outer_values(EllSparseBlockMat<double>& in)
29 {
30  //search outer values in m
31  CooSparseBlockMat<double> out( in.num_rows, 6, in.n, in.left_size, in.right_size);
32  int index = in.data.size()/ in.n/in.n;
33  thrust::host_vector<double> data_element(in.n*in.n, 0), zero(data_element);
34  bool found=false;
35  for( int i=0; i<in.num_rows; i++)
36  for( int d=0; d<in.blocks_per_line; d++)
37  {
38  if( in.cols_idx[i*in.blocks_per_line+d]==-1)
39  { //change the whole line
40  for( int k=0; k<in.blocks_per_line; k++)
41  {
42  for( int j=0; j<in.n*in.n; j++)
43  data_element[j] = in.data[ in.data_idx[i*in.blocks_per_line+k]*in.n*in.n + j];
44  //assume col is either 0,1,or 2
45  int col = in.cols_idx[i*in.blocks_per_line+k] + 1;
46  out.add_value( i, col, data_element);
47  in.data_idx[i*in.blocks_per_line+k] = index; //
48  in.cols_idx[i*in.blocks_per_line+k] = 0;
49  }
50  found=true;
51  }
52  if( in.cols_idx[i*in.blocks_per_line+d]==in.num_cols)
53  {
54  for( int k=0; k<in.blocks_per_line; k++)
55  {
56  for( int j=0; j<in.n*in.n; j++)
57  data_element[j] = in.data[ in.data_idx[i*in.blocks_per_line+k]*in.n*in.n + j];
58  //assume col is either 3,4,or 5
59  int col = in.cols_idx[i*in.blocks_per_line+k] - in.num_cols + 5;
60  out.add_value( i, col, data_element);
61  in.data_idx[i*in.blocks_per_line+k] = index;
62  in.cols_idx[i*in.blocks_per_line+k] = in.num_cols-1;
63  }
64  found=true;
65  }
66  }
67  if(found)
68  {
69  in.data.insert( in.data.end(), zero.begin(), zero.end());
70  }
71  return out;
72 }

83 EllSparseBlockMat<double> distribute_rows( const EllSparseBlockMat<double>& src, int coord, const int* howmany)
84 {
85  if( howmany[1] == 1)
86  {
87  EllSparseBlockMat<double> temp(src);
88  temp.left_size = temp.left_size/howmany[0];
89  temp.right_size = temp.right_size/howmany[2];
90  temp.set_default_range();
91  return temp;
92  }
93  assert( src.num_rows == src.num_cols);
94  int chunk_size = src.num_rows/howmany[1];
95  EllSparseBlockMat<double> temp(chunk_size, chunk_size, src.blocks_per_line, src.data.size()/(src.n*src.n), src.n);
96  temp.left_size = src.left_size/howmany[0];
97  temp.right_size = src.right_size/howmany[2];
98  //first copy data elements (even though not all might be needed it doesn't slow down things either)
99  for( unsigned i=0; i<src.data.size(); i++)
100  temp.data[i] = src.data[i];
101  //now grab the right chunk of cols and data indices
102  for( unsigned i=0; i<temp.cols_idx.size(); i++)
103  {
104  temp.data_idx[i] = src.data_idx[ coord*(chunk_size*src.blocks_per_line)+i];
105  temp.cols_idx[i] = src.cols_idx[ coord*(chunk_size*src.blocks_per_line)+i];
106  //data indices are correct but cols are still the global indices (remapping a bit clumsy)
107  //first in the zeroth line the col idx might be (global)num_cols - 1 -> map that to -1
108  if( coord==0 && i/src.blocks_per_line == 0 && temp.cols_idx[i] == src.num_cols-1) temp.cols_idx[i] = -1;
109  //second in the last line the col idx mighty be 0 -> map to (global)num_cols
110  if( coord==(howmany[1]-1)&& (int)i/src.blocks_per_line == temp.num_rows-1 && temp.cols_idx[i] == 0) temp.cols_idx[i] = src.num_cols;
111  //Elements are now in the range -1, 0, 1,..., (global)num_cols
112  //now shift this range to chunk range -1,..,chunk_size
113  temp.cols_idx[i] = (temp.cols_idx[i] - coord*chunk_size );
114  }
115  temp.set_default_range();
116  return temp;
117 }
