26template<class T>
27T gcd( T a, T b)
28{
29 T r2 = std::max(a,b);
30 T r1 = std::min(a,b);
31 while( r1!=0)
32 {
33 r2 = r2%r1;
34 std::swap( r1, r2);
35 }
36 return r2;

141{
142 if( g_old.N() % g_new.N() != 0) std::cerr << "ATTENTION: you project between incompatible grids!! old N: "<<g_old.N()<<" new N: "<<g_new.N()<<"\n";
143 //form the adjoint
144 cusp::coo_matrix<int, real_type, cusp::host_memory> Wf =
145 dg::create::diagonal( dg::create::weights( g_old));
146 cusp::coo_matrix<int, real_type, cusp::host_memory> Vc =
147 dg::create::diagonal( dg::create::inv_weights( g_new));
148 cusp::coo_matrix<int, real_type, cusp::host_memory> A = interpolationT( g_new, g_old), temp;
150 cusp::multiply( A, Wf, temp);
151 cusp::multiply( Vc, temp, A);
152 A.sort_by_row_and_column();
153 return A;
154}
