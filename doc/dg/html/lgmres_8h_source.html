131 {
132  // Solve for the coefficients, i.e. solve for c in
133  // H*c=s, but we do it in place.
134  int lupe;
135  for (lupe = dimension; lupe >= 0; --lupe)
136  {
137  s[lupe] = s[lupe]/H[lupe][lupe];
138  if(lupe > 0){
139  for (int innerLupe = lupe - 1; innerLupe >= 0; --innerLupe)
140  {
141  // Subtract off the parts from the upper diagonal of the matrix.
142  s[innerLupe] -= s[lupe]*H[innerLupe][lupe];
143  }
144  }
145  }

148  dg::blas1::scal(dx,0.);
