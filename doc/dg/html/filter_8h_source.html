18modal_filter( UnaryOp op, const RealGrid1d<real_type>& g )
19{
20 Operator<real_type> backward=g.dlt().backward();
21 Operator<real_type> forward=g.dlt().forward();
22 Operator<real_type> filter( g.n(), 0);
23 for( unsigned i=0; i<g.n(); i++)
24 filter(i,i) = op( i);
25 filter = backward*filter*forward;
26 //Assemble the matrix
27 EllSparseBlockMat<real_type> A(g.N(), g.N(), 1, 1, g.n());
28 A.data = filter.data();
29 for( unsigned i=0; i<g.N(); i++)
30 {
31 A.data_idx[i] = 0;
32 A.cols_idx[i] = i;
33 }
34 MultiMatrix < EllSparseBlockMat<real_type>, thrust::host_vector<real_type> >
35 filter_matrix(1);
36 filter_matrix.get_matrices()[0] = A;
37 return filter_matrix;
38}
