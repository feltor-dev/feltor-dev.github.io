134{
135 // self-adjoint: apply PCG algorithm to (P 1/W) (W A) x = (P 1/W) (W b) : P' A' x = P' b'
136 // This effectively just replaces all scalar products with the weighted one
137 value_type nrmb = sqrt( blas2::dot( W, b));
138 value_type tol = eps*(nrmb + nrmb_correction);
139#ifdef MPI_VERSION
140 int rank;
141 MPI_Comm_rank(MPI_COMM_WORLD, &rank);
142#endif //MPI
143 if( m_verbose)
144 {
145 DG_RANK0 std::cout << "# Norm of W b "<<nrmb <<"\n";
146 DG_RANK0 std::cout << "# Residual errors: \n";
147 }
148 if( nrmb == 0)
149 {
150 blas1::copy( 0., x);
151 return 0;
152 }
153 blas2::symv( std::forward<Matrix>(A),x,r);
154 blas1::axpby( 1., b, -1., r);
155 if( sqrt( blas2::dot(W,r) ) < tol) //if x happens to be the solution
156 return 0;
157 blas2::symv( std::forward<Preconditioner>(P), r, p );
158 value_type nrmzr_old = blas2::dot( p,W,r); //and store the scalar product
159 value_type alpha, nrmzr_new;
160 for( unsigned i=1; i<max_iter; i++)
161 {
162 blas2::symv( std::forward<Matrix>(A), p, ap);
163 alpha = nrmzr_old/blas2::dot( p, W, ap);
164 blas1::axpby( alpha, p, 1.,x);
165 blas1::axpby( -alpha, ap, 1., r);
166 if( 0 == i%save_on_dots )
167 {
168 if( m_verbose)
169 {
170 DG_RANK0 std::cout << "# Absolute r*W*r "<<sqrt( blas2::dot(W,r)) <<"\t ";
171 DG_RANK0 std::cout << "# < Critical "<<tol <<"\t ";
172 DG_RANK0 std::cout << "# (Relative "<<sqrt( blas2::dot(W,r) )/nrmb << ")\n";
173 }
174 if( sqrt( blas2::dot(W,r)) < tol)
175 return i;
176 }
177 blas2::symv(std::forward<Preconditioner>(P),r,ap);
178 nrmzr_new = blas2::dot( ap, W, r);
179 blas1::axpby(1.,ap, nrmzr_new/nrmzr_old, p );
180 nrmzr_old=nrmzr_new;
181 }
182 if( m_throw_on_fail)
183 {
184 throw dg::Fail( tol, Message(_ping_)
185 <<"After "<<max_iter<<" PCG iterations with rtol "<<eps<<" and atol "<<eps*nrmb_correction );
186 }
