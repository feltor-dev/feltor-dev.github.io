383 template< size_t k, class ContainerType>
384 template< class RHS>
385 void RK_opt<k, ContainerType>::step( RHS& f, real_type t0, const ContainerType& u0, real_type& t1, ContainerType& u1, real_type dt)
386 {
387  f(t0, u0, u_[0]);
388  blas1::axpby( m_rk.alpha[0][0], u0, dt*m_rk.beta[0], u_[0]);
389  std::array<real_type,k-1> tu;
390  tu[0] = dt*m_rk.beta[0];
391  tu[0] = DG_FMA( (m_rk.alpha[0][0]), t0, tu[0]);
392  for( unsigned i=1; i<k-1; i++)
393  {
394  f(tu[i-1], u_[i-1], u_[i] );
395  blas1::axpby( m_rk.alpha[i][0], u0, dt*m_rk.beta[i], u_[i]);
396  tu[i] = dt*m_rk.beta[i];
397  tu[i] = DG_FMA( m_rk.alpha[i][0],t0,tu[i]);
398  for( unsigned l=1; l<=i; l++)
399  {
400  blas1::axpby( m_rk.alpha[i][l], u_[l-1], 1., u_[i]);
401  tu[i] = DG_FMA( m_rk.alpha[i][l], tu[l-1], tu[i]);
402  }
403  }
404  //Now add everything up to u1
405  f(tu[k-2], u_[k-2], u_[k-1]); //u1 may alias u0, so we need u_[k-1]
406  blas1::axpby( m_rk.alpha[k-1][0], u0, dt*m_rk.beta[k-1], u_[k-1], u1);
407  for( unsigned l=1; l<=k-1; l++)
408  blas1::axpby(m_rk.alpha[k-1][l], u_[l-1],1., u1);
409  t1 = t0 + dt;
410 }

480 template< class RHS>
481 void RK<s, ContainerType>::step( RHS& f, real_type t0, const ContainerType& u0, real_type& t1, ContainerType& u1, real_type dt)
482 {
483  f(t0, u0, k_[0]); //compute k_0
484  real_type tu = t0;
485  for( unsigned i=1; i<s; i++) //compute k_i
486  {
487  blas1::axpby( 1., u0, dt*m_rk.a[i][0],k_[0], u_); //l=0
488  tu = DG_FMA( dt,m_rk.a[i][0],t0); //l=0
489  for( unsigned l=1; l<i; l++)
490  {
491  blas1::axpby( dt*m_rk.a[i][l], k_[l],1., u_);
492  tu = DG_FMA(dt,m_rk.a[i][l],tu);
493  }
494  f( tu, u_, k_[i]);
495 
496  }
497  //Now add everything up to u1
498  for( unsigned i=0; i<s; i++)
499  blas1::axpby( dt*m_rk.b[i], k_[i],1., u1);
500  t1 = t0 + dt;

519 template< unsigned s, class RHS, class ContainerType>
520 void stepperRK(RHS& rhs, get_value_type<ContainerType> t_begin, const ContainerType& begin, get_value_type<ContainerType> t_end, ContainerType& end, unsigned N )
521 {
522  using real_type = get_value_type<ContainerType>;
523  RK<s, ContainerType > rk( begin);
524  if( t_end == t_begin){ end = begin; return;}
525  real_type dt = (t_end-t_begin)/(real_type)N;
526  end = begin;
527  real_type t0 = t_begin;
528  for( unsigned i=0; i<N; i++)
529  rk.step( rhs, t0, end, t0, end, dt);
530 }

555 template<unsigned s, class RHS, class ContainerType>
556 int integrateRK(RHS& rhs, get_value_type<ContainerType> t_begin, const ContainerType& begin, get_value_type<ContainerType> t_end, ContainerType& end, get_value_type<ContainerType> eps_abs, unsigned NT_init = 2 )
557 {
558  using real_type = get_value_type<ContainerType>;
559  RK<s, ContainerType > rk( begin);
560  ContainerType old_end(begin);
561  blas1::copy( begin, end );
562  if( t_end == t_begin) return 0;
563  int NT = NT_init;
564  real_type dt = (t_end-t_begin)/(real_type)NT;
565  real_type error = 1e10;
566  real_type t0 = t_begin;
567 
568  while( error > eps_abs && NT < pow( 2, 18) )
569  {
570  blas1::copy( begin, end );
571 
572  int i=0;
573  while (i<NT)
574  {
575  rk.step( rhs, t0, end, t0, end, dt);
576  if( !rhs.monitor( end ) ) //sanity check
577  {
578  #ifdef DG_DEBUG
579  std::cout << "---------Got sanity error -> choosing smaller step size and redo integration" << " NT "<<NT<<" dt "<<dt<< std::endl;
580  #endif
581  break;
582  }
583  i++;
584  }
585  error = rhs.error( end, old_end);
586  blas1::copy( end, old_end);
587  t0 = t_begin;
588  dt /= 2.;
589  NT *= 2;
590  }
591  if( std::isnan( error) )
592  {
593  std::cerr << "ATTENTION: Runge Kutta failed to converge. Error is NAN! "<<std::endl;
594  return -1;
595  }
596  if( error > eps_abs )
597  {
598  std::cerr << "ATTENTION: Runge Kutta failed to converge. Error is "<<error<<" with "<<NT<<" steps"<<std::endl;
599  return -2;
600  }
601  return NT;
