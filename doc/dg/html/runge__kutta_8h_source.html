465 unsigned s = m_rkE.num_stages();
466 value_type tu = t0;
467 //0 stage
469 if( t0 != m_t1)
470 std::get<0>(ode)(t0, u0, m_kE[0]); //freshly compute k_0
471 if( !m_implicit_fsal) // all a(i,0) == 0
472 std::get<1>(ode)(t0, u0, m_kI[0]);
473 // DO NOT HOLD POINTERS AS PRVIATE MEMBERS
474 std::vector<const ContainerType*> k_ptrs( 2*m_rkI.num_stages());
475 for( unsigned i=0; i<m_rkI.num_stages(); i++)
476 {
477 k_ptrs[2*i ] = &m_kE[i];
478 k_ptrs[2*i+1] = &m_kI[i];
479 }
480
481 for( unsigned i=1; i<s; i++)
482 {
483 std::vector<value_type> rka( 2*i);
484 for(unsigned l=0; l<i; l++)
485 {
486 rka[2*l] = m_rkE.a(i,l);
487 rka[2*l+1] = m_rkI.a(i,l);
488 }
489 tu = DG_FMA( m_rkI.c(i),dt, t0);
490 dg::blas1::copy( u0, m_kI[i]);

492 value_type alpha = dt*m_rkI.a(i,i);
493 std::get<2>(ode)( alpha, tu, delta, m_kI[i]);
494 dg::blas1::axpby( 1./alpha, delta, -1./alpha, m_kI[i]);
495 std::get<0>(ode)(tu, delta, m_kE[i]);
496 }
497 m_t1 = t1 = tu;
498 //Now compute result and error estimate
499
500 dg::blas1::copy( u0, u1);
501 detail::gemm( {dt,dt}, dg::asDenseMatrix(k_ptrs), {&m_rkb, &m_rkd},
502 {1.,0.}, {&u1, &delta});
503 //make sure (t1,u1) is the last call to ex
504 std::get<0>(ode)(t1,u1,m_kE[0]);

984template< class ContainerType>
985void SinglestepTimeloop<ContainerType>::do_integrate(
986 value_type& t_begin, const ContainerType&
987 begin, value_type t_end, ContainerType& end,
988 enum to mode ) const
989{
990 bool forward = (t_end - t_begin > 0);
991 if( (m_dt < 0 && forward) || ( m_dt > 0 && !forward) )
992 throw dg::Error( dg::Message(_ping_)<<"Timestep has wrong sign! dt "<<m_dt);
993 dg::blas1::copy( begin, end);
994 if( m_dt == 0)
995 throw dg::Error( dg::Message(_ping_)<<"Timestep may not be zero in SinglestepTimeloop!");
996 if( detail::is_divisable( t_end-t_begin, m_dt))
997 {
998 unsigned N = (unsigned)round((t_end - t_begin)/m_dt);
999 for( unsigned i=0; i<N; i++)
1000 m_step( t_begin, end, t_begin, end, m_dt);
1001 }
1002 else
1003 {
1004 unsigned N = (unsigned)floor( (t_end-t_begin)/m_dt);
1005 for( unsigned i=0; i<N; i++)
1006 m_step( t_begin, end, t_begin, end, m_dt);
1007 if( dg::to::exact == mode)
1008 {
1009 value_type dt_final = t_end - t_begin;
1010 m_step( t_begin, end, t_begin, end, dt_final);
1011 }
1012 else
1013 m_step( t_begin, end, t_begin, end, m_dt);
1014 }
1015 return;
1016}
