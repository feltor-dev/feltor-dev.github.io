812 template<class ContainerType, class SolverType>
813 template< class RHS>
814 void DIRKStep<ContainerType, SolverType>::step( RHS& rhs, value_type t0, const ContainerType& u0, value_type& t1, ContainerType& u1, value_type dt, ContainerType& delta)
815 {
816  unsigned s = m_rkI.num_stages();
817  value_type tu = t0;
818  //0 stage
819  //rhs = u0
820  tu = DG_FMA( m_rkI.c(0),dt, t0);
821  blas1::copy( u0, delta); //better init with rhs
822  if( !(m_rkI.a(0,0)==0) )
823  m_solver.solve( -dt*m_rkI.a(0,0), rhs, tu, delta, u0);
824  rhs(tu, delta, m_kI[0]);
825 
826  //1 stage
827  if( s>1){
828  blas1::evaluate( m_rhs, dg::equals(), PairSum(), 1., u0,
829  dt*m_rkI.a(1,0), m_kI[0]);
830  tu = DG_FMA( m_rkI.c(1),dt, t0);
831  //store solution in delta, init with last solution
832  blas1::copy( m_rhs, delta); //better init with rhs
833  m_solver.solve( -dt*m_rkI.a(1,1), rhs, tu, delta, m_rhs);
834  rhs(tu, delta, m_kI[1]);
835  }
836  //2 stage
837  if( s>2){
838  blas1::evaluate( m_rhs, dg::equals(), PairSum(), 1., u0,
839  dt*m_rkI.a(2,0), m_kI[0],
840  dt*m_rkI.a(2,1), m_kI[1]);
841  tu = DG_FMA( m_rkI.c(2),dt, t0);
842  //just take last solution as init
843  blas1::copy( m_rhs, delta); //better init with rhs
844  m_solver.solve( -dt*m_rkI.a(2,2), rhs, tu, delta, m_rhs);
845  rhs(tu, delta, m_kI[2]);
846  }
847  //3 stage and higher
848  if( s>3){
849  blas1::evaluate( m_rhs, dg::equals(), PairSum(), 1., u0,
850  dt*m_rkI.a(3,0), m_kI[0],
851  dt*m_rkI.a(3,1), m_kI[1],
852  dt*m_rkI.a(3,2), m_kI[2]);
853  tu = DG_FMA( m_rkI.c(3),dt, t0);
854  blas1::copy( m_rhs, delta); //better init with rhs
855  m_solver.solve( -dt*m_rkI.a(3,3), rhs, tu, delta, m_rhs);
856  rhs(tu, delta, m_kI[3]);
857  for( unsigned i=4; i<s; i++)
858  {
859  dg::blas1::copy( u0, m_rhs);
860  for( unsigned j=0; j<i; j++)
861  dg::blas1::axpby( dt*m_rkI.a(i,j), m_kI[j], 1., m_rhs);
862  tu = DG_FMA( m_rkI.c(i),dt, t0);
863  blas1::copy( m_rhs, delta); //better init with rhs
864  m_solver.solve( -dt*m_rkI.a(i,i), rhs, tu, delta, m_rhs);
865  rhs(tu, delta, m_kI[i]);
866  }
867  }
868  t1 = t0 + dt;
869  //Now compute result and error estimate
870  switch( s)
871  {
872  case 1: blas1::subroutine( dg::EmbeddedPairSum(),
873  u1, delta,
874  1., 0., u0,
875  dt*m_rkI.b(0), dt*m_rkI.d(0), m_kI[0]); break;
876  case 2: blas1::subroutine( dg::EmbeddedPairSum(),
877  u1, delta,
878  1., 0., u0,
879  dt*m_rkI.b(0), dt*m_rkI.d(0), m_kI[0],
880  dt*m_rkI.b(1), dt*m_rkI.d(1), m_kI[1]); break;
881  case 3: blas1::subroutine( dg::EmbeddedPairSum(),
882  u1, delta,
883  1., 0., u0,
884  dt*m_rkI.b(0), dt*m_rkI.d(0), m_kI[0],
885  dt*m_rkI.b(1), dt*m_rkI.d(1), m_kI[1],
886  dt*m_rkI.b(2), dt*m_rkI.d(2), m_kI[2]); break;
887  default: blas1::subroutine( dg::EmbeddedPairSum(),
888  u1, delta,
889  1., 0., u0,
890  dt*m_rkI.b(0), dt*m_rkI.d(0), m_kI[0],
891  dt*m_rkI.b(1), dt*m_rkI.d(1), m_kI[1],
892  dt*m_rkI.b(2), dt*m_rkI.d(2), m_kI[2],
893  dt*m_rkI.b(3), dt*m_rkI.d(3), m_kI[3]);
894  //sum the rest
895  for( unsigned i=4; i<s; i++)
896  {
897  dg::blas1::axpby( dt*m_rkI.b(i), m_kI[i], 1., u1);
898  dg::blas1::axpby( dt*m_rkI.d(i), m_kI[i], 1., delta);
899  }
900  }
901 }
