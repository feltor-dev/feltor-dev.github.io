204 {
205  //simplest implementation (all optimization must respect the order of operations)
206  for( int s=0; s<left_size; s++)
207  for( int i=0; i<num_rows; i++)
208  for( int k=0; k<n; k++)
209  for( int j=right_range[0]; j<right_range[1]; j++)
210  {
211  int I = ((s*num_rows + i)*n+k)*right_size+j;
212  y[I]*= beta;
213  for( int d=0; d<blocks_per_line; d++)
214  {
215  value_type temp = 0;
216  for( int q=0; q<n; q++) //multiplication-loop
217  temp = DG_FMA( data[ (data_idx[i*blocks_per_line+d]*n + k)*n+q],
218  x[((s*num_cols + cols_idx[i*blocks_per_line+d])*n+q)*right_size+j],
219  temp);
220  y[I] = DG_FMA( alpha,temp, y[I]);
221  }
