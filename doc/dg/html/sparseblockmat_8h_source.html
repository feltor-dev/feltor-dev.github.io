212 {
213  //simplest implementation (all optimization must respect the order of operations)
214  for( int s=0; s<left_size; s++)
215  for( int i=0; i<num_rows; i++)
216  for( int k=0; k<n; k++)
217  for( int j=right_range[0]; j<right_range[1]; j++)
218  {
219  int I = ((s*num_rows + i)*n+k)*right_size+j;
220  y[I]*= beta;
221  for( int d=0; d<blocks_per_line; d++)
222  {
223  value_type temp = 0;
224  for( int q=0; q<n; q++) //multiplication-loop
225  temp = DG_FMA( data[ (data_idx[i*blocks_per_line+d]*n + k)*n+q],
226  x[((s*num_cols + cols_idx[i*blocks_per_line+d])*n+q)*right_size+j],
227  temp);
228  y[I] = DG_FMA( alpha,temp, y[I]);
229  }
