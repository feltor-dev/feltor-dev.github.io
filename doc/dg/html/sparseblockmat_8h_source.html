229template<class value_type>
230void EllSparseBlockMat<value_type>::symv(SharedVectorTag, SerialTag, value_type alpha, const value_type* RESTRICT x, value_type beta, value_type* RESTRICT y) const
231{
232 //simplest implementation (all optimization must respect the order of operations)
233 for( int s=0; s<left_size; s++)
234 for( int i=0; i<num_rows; i++)
235 for( int k=0; k<n; k++)
236 for( int j=right_range[0]; j<right_range[1]; j++)
237 {
238 int I = ((s*num_rows + i)*n+k)*right_size+j;
239 // if y[I] isnan then even beta = 0 does not make it 0
240 y[I] = beta == 0 ? (value_type)0 : y[I]*beta;
241 for( int d=0; d<blocks_per_line; d++)
242 {
243 value_type temp = 0;
244 for( int q=0; q<n; q++) //multiplication-loop
245 temp = DG_FMA( data[ (data_idx[i*blocks_per_line+d]*n + k)*n+q],
246 x[((s*num_cols + cols_idx[i*blocks_per_line+d])*n+q)*right_size+j],
247 temp);
248 y[I] = DG_FMA( alpha,temp, y[I]);
249 }
250 }
251}
