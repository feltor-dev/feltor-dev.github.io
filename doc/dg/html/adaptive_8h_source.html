37static auto i_control = []( auto dt, auto eps, unsigned embedded_order, unsigned order)
38{
39 using value_type = std::decay_t<decltype(dt[0])>;
40 return dt[0]*pow( eps[0], -1./(value_type)embedded_order);
41};
43static auto pi_control = []( auto dt, auto eps, unsigned embedded_order, unsigned order)
44{
45 using value_type = std::decay_t<decltype(dt[0])>;
46 if( dt[1] == 0)
47 return i_control( dt, eps, embedded_order, order);
48 value_type m_k1 = -0.8, m_k2 = 0.31;
49 value_type factor = pow( eps[0], m_k1/(value_type)embedded_order)
50 * pow( eps[1], m_k2/(value_type)embedded_order);
51 return dt[0]*factor;
52};
71static auto pid_control = []( auto dt, auto eps, unsigned embedded_order, unsigned order)
72{
73 using value_type = std::decay_t<decltype(dt[0])>;
74 if( dt[1] == 0)
75 return i_control( dt, eps, embedded_order, order);
76 if( dt[2] == 0)
77 return pi_control( dt, eps, embedded_order, order);
78 value_type m_k1 = -0.58, m_k2 = 0.21, m_k3 = -0.1;
79 //value_type m_k1 = -0.37, m_k2 = 0.27, m_k3 = -0.1;
80 value_type factor = pow( eps[0], m_k1/(value_type)embedded_order)
81 * pow( eps[1], m_k2/(value_type)embedded_order)
82 * pow( eps[2], m_k3/(value_type)embedded_order);
83 return dt[0]*factor;
84};
85
87static auto ex_control = [](auto dt, auto eps, unsigned embedded_order, unsigned order)
88{
89 using value_type = std::decay_t<decltype(dt[0])>;
90 if( dt[1] == 0)
91 return i_control( dt, eps, embedded_order, order);
92 value_type m_k1 = -0.367, m_k2 = 0.268;
93 value_type factor = pow( eps[0], m_k1/(value_type)embedded_order)
94 * pow( eps[0]/eps[1], m_k2/(value_type)embedded_order);
95 return dt[0]*factor;
96};
98static auto im_control = []( auto dt, auto eps, unsigned embedded_order, unsigned order)
99{
100 using value_type = std::decay_t<decltype(dt[0])>;
101 if( dt[1] == 0)
102 return i_control( dt, eps, embedded_order, order);
103 value_type m_k1 = -0.98, m_k2 = -0.95;
104 value_type factor = pow( eps[0], m_k1/(value_type)embedded_order)
105 * pow( eps[0]/eps[1], m_k2/(value_type)embedded_order);
106 return dt[0]*dt[0]/dt[1]*factor;
107};
109static auto imex_control = []( auto dt, auto eps, unsigned embedded_order, unsigned order)
110{
111 using value_type = std::decay_t<decltype(dt[0])>;
112 value_type h1 = ex_control( dt, eps, embedded_order, order);
113 value_type h2 = im_control( dt, eps, embedded_order, order);
114 // find value closest to zero
115 return fabs( h1) < fabs( h2) ? h1 : h2;
116};

280 container_type& u1,
281 value_type& dt,
282 ControlFunction control,
283 ErrorNorm norm,
284 value_type rtol,
285 value_type atol,
286 value_type reject_limit = 2
287 )
288 {
289 // prevent overwriting u0 in case stepper fails
290 m_stepper.step( std::forward<ODE>(ode), t0, u0, m_t_next, m_next, dt,
291 m_delta);
292 m_nsteps++;
293 dg::blas1::subroutine( detail::Tolerance<value_type>( rtol, atol,
294 m_size), u0, m_delta);
295 m_eps0 = norm( m_delta);
296 m_dt0 = dt;
297 if( m_eps0 > reject_limit || std::isnan( m_eps0) )
298 {
299 // if stepper fails, restart controller
300 dt = control( std::array<value_type,3>{m_dt0, 0, m_dt2},
301 std::array<value_type,3>{m_eps0, m_eps1, m_eps2},
302 m_stepper.embedded_order(),
303 m_stepper.order());
304 if( fabs( dt) > 0.9*fabs(m_dt0))
305 dt = 0.9*m_dt0;
306 //0.9*m_dt0 is a safety limit
307 //that prevents an increase of the timestep in case the stepper fails
308 m_failed = true; m_nfailed++;
309 dg::blas1::copy( u0, u1);
310 t1 = t0;
311 return;
312 }
313 if( m_eps0 < 1e-20) // small or zero
314 {
315 dt = 1e14*m_dt0; // a very large number
316 m_eps0 = 1e-20; // prevent storing zero
317 }
318 else
319 {
320 dt = control( std::array<value_type,3>{m_dt0, m_dt1, m_dt2},
321 std::array<value_type,3>{m_eps0, m_eps1, m_eps2},
322 m_stepper.embedded_order(),
323 m_stepper.order());
324 }
325 m_eps2 = m_eps1;
326 m_eps1 = m_eps0;
327 m_dt2 = m_dt1;
328 m_dt1 = m_dt0;
329 dg::blas1::copy( m_next, u1);
330 t1 = m_t_next;
331 m_failed = false;

618template< class ContainerType>
619void AdaptiveTimeloop<ContainerType>::do_integrate(
620 value_type t0,
621 const ContainerType& u0,
622 value_type& t1,
623 ContainerType& u1,
624 bool check
625 )const
626{
627 value_type t_current = t0;
628 blas1::copy( u0, u1 );
629 bool forward = (t1 - t0 > 0);
630 value_type& dt_current = m_dt_current.data();
631 if( dt_current == 0)
632 dt_current = forward ? 1e-6 : -1e-6; // a good a guess as any
633
634 while( (forward && t_current < t1) || (!forward && t_current > t1))
635 {
636 if( check &&( (forward && t_current+dt_current > t1) || (!forward &&
637 t_current + dt_current < t1) ) )
638 dt_current = t1-t_current;
639 // Compute a step and error
640 try{
641 m_step( t_current, u1, t_current, u1, dt_current);
642 }catch ( dg::Error& e)
643 {
644 e.append( dg::Message(_ping_) << "Error in AdaptiveTimeloop::integrate");
645 throw;
646 }
647 if( !std::isfinite(dt_current) || fabs(dt_current) < 1e-9*fabs(t1-t0))
648 {
649 value_type dt_current0 = dt_current;
650 dt_current = 0.;
651 throw dg::Error(dg::Message(_ping_)<<"Adaptive integrate failed to converge! dt = "<<std::scientific<<dt_current0);
652 }
653 }

656template< class ContainerType>
657template< class Domain>
658void AdaptiveTimeloop<ContainerType>::integrate_in_domain(
659 value_type t0,
660 const ContainerType& u0,
661 value_type& t1,
662 ContainerType& u1,
663 value_type dt,
664 Domain&& domain,
665 value_type eps_root
666 )
667{
668 value_type t_current = t0, dt_current = dt;
669 blas1::copy( u0, u1 );
670 ContainerType& current(u1);
671 if( t1 == t0)
672 return;
673 bool forward = (t1 - t0 > 0);
674 if( dt == 0)
675 dt_current = forward ? 1e-6 : -1e-6; // a good a guess as any
676
677 ContainerType last( u0);
678 while( (forward && t_current < t1) || (!forward && t_current > t1))
679 {
680 //remember last step
681 t0 = t_current;
682 dg::blas1::copy( current, last);
683 if( (forward && t_current+dt_current > t1) || (!forward && t_current +
684 dt_current < t1) )
685 dt_current = t1-t_current;
686 // Compute a step and error
687 try{
688 m_step( t_current, current, t_current, current, dt_current);
689 }catch ( dg::Error& e)
690 {
691 e.append( dg::Message(_ping_) << "Error in AdaptiveTimeloop::integrate");
692 throw;
693 }
694 if( !std::isfinite(dt_current) || fabs(dt_current) < 1e-9*fabs(t1-t0))
695 throw dg::Error(dg::Message(_ping_)<<"integrate_in_domain failed to converge! dt = "<<std::scientific<<dt_current);
696 if( !domain.contains( current) )
697 {
698 //start bisection between t0 and t1
699 t1 = t_current;
700
701 // t0 and last are inside
702 dg::blas1::copy( last, current);
703 int j_max = 50;
704 for(int j=0; j<j_max; j++)
705 {
706 if( fabs(t1-t0) < eps_root*fabs(t1) + eps_root)
707 {
708 return;
709 }
710 dt_current = (t1-t0)/2.;
711 t_current = t0;
712 // Here we can assume that the timestepper always succeeds
713 // because dt_current is always smaller than the previous timestep
714 // t_current = t0, current = last (inside)
715 m_step( t_current, current, t_current, current, dt_current);
716 //stepper( t0, last, t_middle, u1, (t1-t0)/2.);

718 {
719 t0 = t_current;
720 dg::blas1::copy( current, last);
721 }
722 else
723 {
724 t1 = t_current;
725 dg::blas1::copy( last, current);
726 }
727 }
