51 std::vector<real_type> coefficients( real_type xn, unsigned n)
52 {
53  assert( xn <= 1. && xn >= -1.);
54  std::vector<real_type> px(n);
55  if( xn == -1)
56  {
57  for( unsigned u=0; u<n; u++)
58  px[u] = (u%2 == 0) ? +1. : -1.;
59  }
60  else if( xn == 1)
61  {
62  for( unsigned i=0; i<n; i++)
63  px[i] = 1.;
64  }
65  else
66  {
67  px[0] = 1.;
68  if( n > 1)
69  {
70  px[1] = xn;
71  for( unsigned i=1; i<n-1; i++)
72  px[i+1] = ((real_type)(2*i+1)*xn*px[i]-(real_type)i*px[i-1])/(real_type)(i+1);
73  }
74  }
75  return px;
76 }
