45 {
46  assert( xn <= 1. && xn >= -1.);

48  if( xn == -1)
49  {
50  for( unsigned i=0; i<n; i++)
51  px[i] = (real_type)pow( -1, i);
52  }
53  else if( xn == 1)
54  {
55  for( unsigned i=0; i<n; i++)
56  px[i] = 1.;
57  }
58  else
59  {
60  px[0] = 1.;

62  {
63  px[1] = xn;
64  for( unsigned i=1; i<n-1; i++)

67  }
