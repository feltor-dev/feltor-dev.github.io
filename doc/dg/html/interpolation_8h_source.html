54template<class real_type>
55std::vector<real_type> coefficients( real_type xn, unsigned n)
56{
57 assert( xn <= 1. && xn >= -1.);
58 std::vector<real_type> px(n);
59 if( xn == -1)
60 {
61 for( unsigned u=0; u<n; u++)
62 px[u] = (u%2 == 0) ? +1. : -1.;
63 }
64 else if( xn == 1)
65 {
66 for( unsigned i=0; i<n; i++)
67 px[i] = 1.;
68 }
69 else
70 {
71 px[0] = 1.;
72 if( n > 1)
73 {
74 px[1] = xn;
75 for( unsigned i=1; i<n-1; i++)
76 px[i+1] = ((real_type)(2*i+1)*xn*px[i]-(real_type)i*px[i-1])/(real_type)(i+1);
77 }
78 }
79 return px;
80}

83template<class real_type>
84std::vector<real_type> lagrange( real_type x, const std::vector<real_type>& xi)
85{
86 unsigned n = xi.size();
87 std::vector<real_type> l( n , 1.);
88 for( unsigned i=0; i<n; i++)
89 for( unsigned k=0; k<n; k++)
90 {
91 if ( k != i)
92 l[i] *= (x-xi[k])/(xi[i]-xi[k]);
93 }
94 return l;
95}

98template<class real_type>
99std::vector<real_type> choose_1d_abscissas( real_type X,
100 unsigned points_per_line, const RealGrid1d<real_type>& g,
101 const thrust::host_vector<real_type>& abs,
102 thrust::host_vector<unsigned>& cols)
103{
104 assert( abs.size() >= points_per_line && "There must be more points to interpolate\n");
105 dg::bc bcx = g.bcx();
106 //determine which cell (X) lies in
107 real_type xnn = (X-g.x0())/g.h();
108 unsigned n = (unsigned)floor(xnn);
109 //intervall correction
110 if (n==g.N() && bcx != dg::PER) {
111 n-=1;
112 }
113 // look for closest abscissa
114 std::vector<real_type> xs( points_per_line, 0);
115 // X <= *it
116 auto it = std::lower_bound( abs.begin()+n*g.n(), abs.begin() + (n+1)*g.n(),
117 X);
118 cols.resize( points_per_line, 0);
119 switch( points_per_line)
120 {
121 case 1: xs[0] = 1.;
122 if( it == abs.begin())
123 cols[0] = 0;
124 else if( it == abs.end())
125 cols[0] = it - abs.begin() - 1;
126 else
127 {
128 if ( fabs(X - *it) < fabs( X - *(it-1)))
129 cols[0] = it - abs.begin();
130 else
131 cols[0] = it - abs.begin() -1;
132 }
133 break;
134 case 2: if( it == abs.begin())
135 {
136 if( bcx == dg::PER)
137 {
138 xs[0] = *it;
139 xs[1] = *(abs.end() -1)-g.lx();;
140 cols[0] = 0, cols[1] = abs.end()-abs.begin()-1;
141 }
142 else
143 {
144 //xs[0] = *it;
145 //xs[1] = *(it+1);
146 //cols[0] = 0, cols[1] = 1;
147 // This makes it consistent with fem_t
148 xs.resize(1);
149 xs[0] = *it;
150 cols[0] = 0;
151 }
152 }
153 else if( it == abs.end())
154 {
155 if( bcx == dg::PER)
156 {
157 xs[0] = *(abs.begin())+g.lx();
158 xs[1] = *(it-1);
159 cols[0] = 0, cols[1] = it-abs.begin()-1;
160 }
161 else
162 {
163 //xs[0] = *(it-2);
164 //xs[0] = *(it-1);
165 //cols[0] = it - abs.begin() - 2;
166 //cols[1] = it - abs.begin() - 1;
167 // This makes it consistent with fem_t
168 xs.resize(1);
169 xs[0] = *(it-1);
170 cols[0] = it-abs.begin()-1;
171 }
172 }
173 else
174 {
175 xs[0] = *(it-1);
176 xs[1] = *it;
177 cols[0] = it - abs.begin() - 1;
178 cols[1] = cols[0]+1;
179 }
180 break;
181 case 4: if( it <= abs.begin() +1)
182 {
183 if( bcx == dg::PER)
184 {
185 xs[0] = *abs.begin(), cols[0] = 0;
186 xs[1] = *(abs.begin()+1), cols[1] = 1;
187 xs[2] = it == abs.begin() ? *(abs.end() -2) : *(abs.begin()+2);
188 cols[2] = it == abs.begin() ? abs.end()-abs.begin() -2 : 2;
189 xs[3] = *(abs.end() -1);
190 cols[3] = abs.end()-abs.begin() -1;
191 }
192 else
193 {
194 it = abs.begin();
195 xs[0] = *it, xs[1] = *(it+1);
196 xs[2] = *(it+2), xs[3] = *(it+3);
197 cols[0] = 0, cols[1] = 1;
198 cols[2] = 2, cols[3] = 3;
199 }
200 }
201 else if( it >= abs.end() -1)
202 {
203 if( bcx == dg::PER)
204 {
205 xs[0] = *abs.begin(), cols[0] = 0;
206 xs[1] = it == abs.end() ? *(abs.begin()+1) : *(abs.end() -3) ;
207 cols[1] = it == abs.end() ? 1 : abs.end()-abs.begin()-3 ;
208 xs[2] = *(abs.end() - 2), cols[2] = abs.end()-abs.begin()-2;
209 xs[3] = *(abs.end() - 1), cols[3] = abs.end()-abs.begin()-1;
210 }
211 else
212 {
213 it = abs.end();
214 xs[0] = *(it-4), xs[1] = *(it-3);
215 xs[2] = *(it-2), xs[3] = *(it-1);
216 cols[0] = it - abs.begin() - 4;
217 cols[1] = cols[0]+1;
218 cols[2] = cols[1]+1;
219 cols[3] = cols[2]+1;
220 }
221 }
222 else
223 {
224 xs[0] = *(it-2), xs[1] = *(it-1);
225 xs[2] = *(it ), xs[3] = *(it+1);
226 cols[0] = it - abs.begin() - 2;
227 cols[1] = cols[0]+1;
228 cols[2] = cols[1]+1;
229 cols[3] = cols[2]+1;
230 }
231 break;
232 }
233 return xs;
234}

274template<class real_type>
275cusp::coo_matrix<int, real_type, cusp::host_memory> interpolation(
276 const thrust::host_vector<real_type>& x,
277 const RealGrid1d<real_type>& g,
278 dg::bc bcx = dg::NEU,
279 std::string method = "dg")
280{
281 cusp::array1d<real_type, cusp::host_memory> values;
282 cusp::array1d<int, cusp::host_memory> row_indices;
283 cusp::array1d<int, cusp::host_memory> column_indices;
284 if( method == "dg")
285 {
286 dg::Operator<real_type> forward( g.dlt().forward());
287 for( unsigned i=0; i<x.size(); i++)
288 {
289 real_type X = x[i];
290 bool negative = false;
291 g.shift( negative, X, bcx);
292
293 //determine which cell (x) lies in
294 real_type xnn = (X-g.x0())/g.h();
295 unsigned n = (unsigned)floor(xnn);
296 //determine normalized coordinates
297 real_type xn = 2.*xnn - (real_type)(2*n+1);
298 //intervall correction
299 if (n==g.N()) {
300 n-=1;
301 xn = 1.;
302 }
303 //evaluate 2d Legendre polynomials at (xn, yn)...
304 std::vector<real_type> px = detail::coefficients( xn, g.n());
305 //...these are the matrix coefficients with which to multiply
306 std::vector<real_type> pxF(px.size(),0);
307 for( unsigned l=0; l<g.n(); l++)
308 for( unsigned k=0; k<g.n(); k++)
309 pxF[l]+= px[k]*forward(k,l);
310 unsigned cols = n*g.n();
311 for ( unsigned l=0; l<g.n(); l++)
312 {
313 row_indices.push_back(i);
314 column_indices.push_back( cols + l);
315 values.push_back(negative ? -pxF[l] : pxF[l]);
316 }
317 }
318 }
319 else
320 {
321 unsigned points_per_line = 1;
322 if( method == "nearest")
323 points_per_line = 1;
324 else if( method == "linear")
325 points_per_line = 2;
326 else if( method == "cubic")
327 points_per_line = 4;
328 else
329 throw std::runtime_error( "Interpolation method "+method+" not recognized!\n");
330 thrust::host_vector<real_type> abs = dg::create::abscissas( g);
331 dg::RealGrid1d<real_type> gx( g.x0(), g.x1(), g.n(), g.N(), bcx);
332 for( unsigned i=0; i<x.size(); i++)
333 {
334 real_type X = x[i];
335 bool negative = false;
336 g.shift( negative, X, bcx);
337
338 thrust::host_vector<unsigned> cols;
339 std::vector<real_type> xs = detail::choose_1d_abscissas( X,
340 points_per_line, gx, abs, cols);
341
342 std::vector<real_type> px = detail::lagrange( X, xs);
343 // px may have size != points_per_line (at boundary)
344 for ( unsigned l=0; l<px.size(); l++)
345 {
346 row_indices.push_back(i);
347 column_indices.push_back( cols[l]);
348 values.push_back(negative ? -px[l] : px[l]);
349 }
350 }
351 }
352 cusp::coo_matrix<int, real_type, cusp::host_memory> A(
353 x.size(), g.size(), values.size());
354 A.row_indices = row_indices;
355 A.column_indices = column_indices;
356 A.values = values;
357 return A;
358}

387 std::string method = "dg")

390 cusp::array1d<real_type, cusp::host_memory> values;
391 cusp::array1d<int, cusp::host_memory> row_indices;
392 cusp::array1d<int, cusp::host_memory> column_indices;
393 if( method == "dg")
394 {
395 std::vector<real_type> gauss_nodesx = g.dltx().abscissas();
396 std::vector<real_type> gauss_nodesy = g.dlty().abscissas();
397 dg::Operator<real_type> forwardx( g.dltx().forward());
398 dg::Operator<real_type> forwardy( g.dlty().forward());

403 real_type X = x[i], Y = y[i];
404 bool negative=false;

407 //determine which cell (x,y) lies in
408 real_type xnn = (X-g.x0())/g.hx();
409 real_type ynn = (Y-g.y0())/g.hy();
410 unsigned nn = (unsigned)floor(xnn);
411 unsigned mm = (unsigned)floor(ynn);
412 //determine normalized coordinates
413 real_type xn = 2.*xnn - (real_type)(2*nn+1);
414 real_type yn = 2.*ynn - (real_type)(2*mm+1);
415 //interval correction
416 if (nn==g.Nx()) {
417 nn-=1;
418 xn =1.;
419 }
420 if (mm==g.Ny()) {
421 mm-=1;
422 yn =1.;
423 }
424 //Test if the point is a Gauss point since then no interpolation is needed
425 int idxX =-1, idxY = -1;
426 for( unsigned k=0; k<g.nx(); k++)

429 idxX = nn*g.nx() + k; //determine which grid column it is
430 }
431 for( unsigned k=0; k<g.ny(); k++)
432 {
433 if( fabs( yn - gauss_nodesy[k]) < 1e-14)
434 idxY = mm*g.ny() + k; //determine grid line
435 }
436 if( idxX < 0 && idxY < 0 ) //there is no corresponding point
437 {
438 //evaluate 2d Legendre polynomials at (xn, yn)...
439 std::vector<real_type> px = detail::coefficients( xn, g.nx()),
440 py = detail::coefficients( yn, g.ny());
441 std::vector<real_type> pxF(g.nx(),0), pyF(g.ny(), 0);
442 for( unsigned l=0; l<g.nx(); l++)
443 for( unsigned k=0; k<g.nx(); k++)
444 pxF[l]+= px[k]*forwardx(k,l);
445 for( unsigned l=0; l<g.ny(); l++)
446 for( unsigned k=0; k<g.ny(); k++)
447 pyF[l]+= py[k]*forwardy(k,l);
448 //these are the matrix coefficients with which to multiply
449 for(unsigned k=0; k<g.ny(); k++)
450 for( unsigned l=0; l<g.nx(); l++)
451 {
452 row_indices.push_back( i);
453 column_indices.push_back( ((mm*g.ny()+k)*g.Nx()+nn)*g.nx() + l);
454 real_type pxy = pyF[k]*pxF[l];
455 if( !negative)
456 values.push_back( pxy);
457 else
458 values.push_back( -pxy);
459 }
460 }
461 else if ( idxX < 0 && idxY >=0) //there is a corresponding line
462 {
463 std::vector<real_type> px = detail::coefficients( xn, g.nx());
464 std::vector<real_type> pxF(g.nx(),0);
465 for( unsigned l=0; l<g.nx(); l++)
466 for( unsigned k=0; k<g.nx(); k++)
467 pxF[l]+= px[k]*forwardx(k,l);
468 for( unsigned l=0; l<g.nx(); l++)
469 {
470 row_indices.push_back( i);
471 column_indices.push_back( ((idxY)*g.Nx() + nn)*g.nx() + l);
472 if( !negative)
473 values.push_back( pxF[l]);
474 else
475 values.push_back(-pxF[l]);
476
477 }
478 }
479 else if ( idxX >= 0 && idxY < 0) //there is a corresponding column
480 {
481 std::vector<real_type> py = detail::coefficients( yn, g.ny());
482 std::vector<real_type> pyF(g.ny(),0);
483 for( unsigned l=0; l<g.ny(); l++)
484 for( unsigned k=0; k<g.ny(); k++)
485 pyF[l]+= py[k]*forwardy(k,l);
486 for( unsigned k=0; k<g.ny(); k++)
487 {
488 row_indices.push_back(i);
489 column_indices.push_back((mm*g.ny()+k)*g.Nx()*g.nx() + idxX);
490 if( !negative)
491 values.push_back( pyF[k]);
492 else
493 values.push_back(-pyF[k]);
494
495 }
496 }
497 else //the point already exists
498 {
499 row_indices.push_back(i);
500 column_indices.push_back(idxY*g.Nx()*g.nx() + idxX);
501 if( !negative)
502 values.push_back( 1.);
503 else
504 values.push_back(-1.);
505 }
506
507 }
508 }
509 else
510 {
511 unsigned points_per_line = 1;
512 if( method == "nearest")
513 points_per_line = 1;
514 else if( method == "linear")
515 points_per_line = 2;
516 else if( method == "cubic")
517 points_per_line = 4;
518 else
519 throw std::runtime_error( "Interpolation method "+method+" not recognized!\n");
520 RealGrid1d<real_type> gx(g.x0(), g.x1(), g.nx(), g.Nx(), bcx);
521 RealGrid1d<real_type> gy(g.y0(), g.y1(), g.ny(), g.Ny(), bcy);
522 thrust::host_vector<real_type> absX = dg::create::abscissas( gx);
523 thrust::host_vector<real_type> absY = dg::create::abscissas( gy);
524
525 for( unsigned i=0; i<x.size(); i++)
526 {
527 real_type X = x[i], Y = y[i];
528 bool negative = false;
529 g.shift( negative, X, Y, bcx, bcy);
530
531 thrust::host_vector<unsigned> colsX, colsY;
532 std::vector<real_type> xs = detail::choose_1d_abscissas( X,
533 points_per_line, gx, absX, colsX);
534 std::vector<real_type> ys = detail::choose_1d_abscissas( Y,
535 points_per_line, gy, absY, colsY);

538 std::vector<real_type> pxy( points_per_line*points_per_line);
539 std::vector<real_type> px = detail::lagrange( X, xs),
540 py = detail::lagrange( Y, ys);
541 // note: px , py may have size != points_per_line at boundary
542 for(unsigned k=0; k<py.size(); k++)
543 for( unsigned l=0; l<px.size(); l++)
544 pxy[k*px.size()+l]= py[k]*px[l];
545 for( unsigned k=0; k<py.size(); k++)
546 for( unsigned l=0; l<px.size(); l++)
547 {
548 if( fabs(pxy[k*px.size() +l]) > 1e-14)
549 {
550 row_indices.push_back( i);
551 column_indices.push_back( (colsY[k])*g.nx()*g.Nx() +
552 colsX[l]);
553 values.push_back( negative ? - pxy[k*px.size()+l]
554 : pxy[k*px.size()+l]);
555 }
556 }
557 }
558 }
559 cusp::coo_matrix<int, real_type, cusp::host_memory> A( x.size(),
560 g.size(), values.size());
561 A.row_indices = row_indices;
562 A.column_indices = column_indices;
563 A.values = values;
564
565 return A;
566}

698 {
699 row_indices.push_back(i);
700 column_indices.push_back(((idxZ*g.Ny()+mm)*g.ny()+k)*g.Nx()*g.nx() + idxX);
701 if(!negative)
702 values.push_back( pyF[k]);
703 else
704 values.push_back(-pyF[k]);
705 }
706 }
707 else
708 {
709 //evaluate 3d Legendre polynomials at (xn, yn, zn)...
710 std::vector<real_type> px = detail::coefficients( xn, g.nx()),
711 py = detail::coefficients( yn, g.ny()),
712 pz = detail::coefficients( zn, g.nz());
713 std::vector<real_type> pxF(g.nx(),0), pyF(g.ny(), 0), pzF( g.nz(), 0);
714 for( unsigned l=0; l<g.nx(); l++)
715 for( unsigned k=0; k<g.nx(); k++)
716 pxF[l]+= px[k]*forwardx(k,l);
717 for( unsigned l=0; l<g.ny(); l++)
718 for( unsigned k=0; k<g.ny(); k++)
719 pyF[l]+= py[k]*forwardy(k,l);
720 for( unsigned l=0; l<g.nz(); l++)
721 for( unsigned k=0; k<g.nz(); k++)
722 pzF[l]+= pz[k]*forwardz(k,l);
723 //these are the matrix coefficients with which to multiply
724 for( unsigned s=0; s<g.nz(); s++)
725 for( unsigned k=0; k<g.ny(); k++)
726 for( unsigned l=0; l<g.nx(); l++)
727 {
728 row_indices.push_back( i);
729 column_indices.push_back(
730 ((((ll*g.nz()+s)*g.Ny()+mm)*g.ny()+k)*g.Nx()+nn)*g.nx()+l);
731 real_type pxyz = pzF[s]*pyF[k]*pxF[l];
732 if( !negative)
733 values.push_back( pxyz);
734 else
735 values.push_back(-pxyz);
736 }
737 }
738 }
739 }
740 else
741 {
742 unsigned points_per_line = 1;
743 if( method == "nearest")
744 points_per_line = 1;
745 else if( method == "linear")
746 points_per_line = 2;
747 else if( method == "cubic")
748 points_per_line = 4;
749 else
750 throw std::runtime_error( "Interpolation method "+method+" not recognized!\n");
751 RealGrid1d<real_type> gx(g.x0(), g.x1(), g.nx(), g.Nx(), bcx);
752 RealGrid1d<real_type> gy(g.y0(), g.y1(), g.ny(), g.Ny(), bcy);
753 RealGrid1d<real_type> gz(g.z0(), g.z1(), g.nz(), g.Nz(), bcz);
754 thrust::host_vector<real_type> absX = dg::create::abscissas( gx);
755 thrust::host_vector<real_type> absY = dg::create::abscissas( gy);
756 thrust::host_vector<real_type> absZ = dg::create::abscissas( gz);
757 for( unsigned i=0; i<x.size(); i++)
758 {
759 real_type X = x[i], Y = y[i], Z = z[i];
760 bool negative = false;
761 g.shift( negative, X, Y, Z, bcx, bcy, bcz);
762
763 thrust::host_vector<unsigned> colsX, colsY, colsZ;
764 std::vector<real_type> xs = detail::choose_1d_abscissas( X,
765 points_per_line, gx, absX, colsX);
766 std::vector<real_type> ys = detail::choose_1d_abscissas( Y,
767 points_per_line, gy, absY, colsY);
768 std::vector<real_type> zs = detail::choose_1d_abscissas( Z,
769 points_per_line, gz, absZ, colsZ);
770
771 //evaluate 3d Legendre polynomials at (xn, yn, zn)...
772 std::vector<real_type> pxyz( points_per_line*points_per_line
773 *points_per_line);
774 std::vector<real_type> px = detail::lagrange( X, xs),
775 py = detail::lagrange( Y, ys),
776 pz = detail::lagrange( Z, zs);
777 // note: px, py, pz may have size != points_per_line at boundary
778 for( unsigned m=0; m<pz.size(); m++)
779 for( unsigned k=0; k<py.size(); k++)
780 for( unsigned l=0; l<px.size(); l++)
781 pxyz[(m*py.size()+k)*px.size()+l]= pz[m]*py[k]*px[l];
782 for( unsigned m=0; m<pz.size(); m++)
783 for( unsigned k=0; k<py.size(); k++)
784 for( unsigned l=0; l<px.size(); l++)
785 {
786 if( fabs(pxyz[(m*py.size()+k)*px.size() +l]) > 1e-14)
787 {
788 row_indices.push_back( i);
789 column_indices.push_back( ((colsZ[m])*g.ny()*g.Ny() +
790 colsY[k])*g.nx()*g.Nx() + colsX[l]);
791 values.push_back( negative ?
792 -pxyz[(m*py.size()+k)*px.size()+l]
793 : pxyz[(m*py.size()+k)*px.size()+l] );
794 }
795 }
796 }
797 }
798 cusp::coo_matrix<int, real_type, cusp::host_memory> A( x.size(), g.size(),
799 values.size());
800 A.row_indices = row_indices;
801 A.column_indices = column_indices;
802 A.values = values;
803
804 return A;
805}

895thrust::host_vector<real_type> forward_transform( const thrust::host_vector<real_type>& in, const aRealTopology2d<real_type>& g)

899 dg::Operator<real_type> forwardy( g.dlty().forward());
900 for( unsigned i=0; i<g.Ny(); i++)
901 for( unsigned k=0; k<g.ny(); k++)
902 for( unsigned j=0; j<g.Nx(); j++)
903 for( unsigned l=0; l<g.nx(); l++)

929template<class real_type>
930real_type interpolate(
931 dg::space sp,
932 const thrust::host_vector<real_type>& v,
933 real_type x,
934 const RealGrid1d<real_type>& g,
935 dg::bc bcx = dg::NEU)
936{
937 assert( v.size() == g.size());
938 bool negative = false;
939 g.shift( negative, x, bcx);
940
941 //determine which cell (x) lies in
942
943 real_type xnn = (x-g.x0())/g.h();
944 unsigned n = (unsigned)floor(xnn);
945 //determine normalized coordinates
946
947 real_type xn = 2.*xnn - (real_type)(2*n+1);
948 //interval correction
949 if (n==g.N()) {
950 n-=1;
951 xn = 1.;
952 }
953 //evaluate 1d Legendre polynomials at (xn)...
954 std::vector<real_type> px = create::detail::coefficients( xn, g.n());
955 if( sp == dg::xspace)
956 {
957 dg::Operator<real_type> forward( g.dlt().forward());
958 std::vector<real_type> pxF(g.n(),0);
959 for( unsigned l=0; l<g.n(); l++)
960 for( unsigned k=0; k<g.n(); k++)
961 pxF[l]+= px[k]*forward(k,l);
962 for( unsigned k=0; k<g.n(); k++)
963 px[k] = pxF[k];
964 }
965 //these are the matrix coefficients with which to multiply
966 unsigned cols = (n)*g.n();
967 //multiply x
968 real_type value = 0;
969 for( unsigned j=0; j<g.n(); j++)
970 {
971 if(negative)
972 value -= v[cols + j]*px[j];
973 else
974 value += v[cols + j]*px[j];
975 }
976 return value;
977}

998template<class real_type>
999real_type interpolate(
1000 dg::space sp,
1001 const thrust::host_vector<real_type>& v,
1002 real_type x, real_type y,
1003 const aRealTopology2d<real_type>& g,
1004 dg::bc bcx = dg::NEU, dg::bc bcy = dg::NEU )
1005{
1006 assert( v.size() == g.size());
1007 bool negative = false;
1008 g.shift( negative, x,y, bcx, bcy);
1009
1010 //determine which cell (x,y) lies in
1011
1012 real_type xnn = (x-g.x0())/g.hx();
1013 real_type ynn = (y-g.y0())/g.hy();
1014 unsigned n = (unsigned)floor(xnn);
1015 unsigned m = (unsigned)floor(ynn);
1016 //determine normalized coordinates
1017
1018 real_type xn = 2.*xnn - (real_type)(2*n+1);
1019 real_type yn = 2.*ynn - (real_type)(2*m+1);
1020 //interval correction
1021 if (n==g.Nx()) {
1022 n-=1;
1023 xn = 1.;
1024 }
1025 if (m==g.Ny()) {
1026 m-=1;
1027 yn =1.;
1028 }
1029 //evaluate 2d Legendre polynomials at (xn, yn)...
1030 std::vector<real_type> px = create::detail::coefficients( xn, g.nx()),
1031 py = create::detail::coefficients( yn, g.ny());
1032 if( sp == dg::xspace)
1033 {
1034 dg::Operator<real_type> forwardx( g.dltx().forward());
1035 dg::Operator<real_type> forwardy( g.dlty().forward());
1036 std::vector<real_type> pxF(g.nx(),0), pyF(g.ny(), 0);
1037 for( unsigned l=0; l<g.nx(); l++)
1038 for( unsigned k=0; k<g.nx(); k++)
1039 pxF[l]+= px[k]*forwardx(k,l);
1040 for( unsigned l=0; l<g.ny(); l++)
1041 for( unsigned k=0; k<g.ny(); k++)
1042 pyF[l]+= py[k]*forwardy(k,l);
1043 px = pxF, py = pyF;
1044 }
1045 //these are the matrix coefficients with which to multiply
1046 unsigned cols = (m)*g.Nx()*g.ny()*g.nx() + (n)*g.nx();
1047 //multiply x
1048 real_type value = 0;
1049 for( unsigned i=0; i<g.ny(); i++)
1050 for( unsigned j=0; j<g.nx(); j++)
1051 {
1052 if(negative)
1053 value -= v[cols + i*g.Nx()*g.nx() + j]*px[j]*py[i];
1054 else
1055 value += v[cols + i*g.Nx()*g.nx() + j]*px[j]*py[i];
1056 }
1057 return value;
1058}
