28template<class real_type>
29thrust::host_vector<real_type> normalize_weights_and_compute_abscissas( const RealGrid1d<real_type>& g, thrust::host_vector<real_type>& weights)
30{
31 //normalize weights
32 unsigned Nx_new = weights.size()/g.n();
33 for( unsigned i=0;i<weights.size(); i++)
34 weights[i] *= (real_type)g.N()/(real_type)Nx_new;
35
36 thrust::host_vector<real_type> boundaries(Nx_new+1), abs(g.n()*Nx_new);
37 boundaries[0] = g.x0();
38 for( unsigned i=0; i<Nx_new; i++)
39 {
40 boundaries[i+1] = boundaries[i] + g.lx()/(real_type)Nx_new/weights[g.n()*i];
41 for( unsigned j=0; j<g.n(); j++)
42 {
43 abs[i*g.n()+j] = (boundaries[i+1]+boundaries[i])/2. +
44 (boundaries[i+1]-boundaries[i])/2.*g.dlt().abscissas()[j];
45 }
46 }
47 return abs;
48}

164 std::vector<real_type> simpsons_weights( unsigned N, double h) const
165 {
166 // Create weights for Simpson's integration on an equidistant subgrid with N cells (including end points)
167 // N >= 2
168 std::vector<real_type> simpson( N, h);
169 if( N == 2)
170 {
171 simpson[0] = simpson[1] = h/2.;
172 return simpson;
173 }
174 simpson[0]=1./3. * h;
175 for( unsigned i=0; i<(N-3)/2; i++)
176 {
177 simpson[2*i+1] = 4./3. * h;
178 simpson[2*i+2] = 2./3. * h;
179 }
180 if( N%2 != 0)
181 {
182 simpson[N-2] = 4./3. * h;
183 simpson[N-1] = 1./3. * h;
184 }
185 else
186 {
187 simpson[N-4] = N==4 ? 3./8.*h : (1./3.+3./8.)*h;
188 simpson[N-3] = (9./8.)*h;
189 simpson[N-2] = (9./8.)*h;
190 simpson[N-1] = (3./8.)*h;
191 }
192 return simpson;
193 }
194 virtual void do_generate( const RealGrid1d<real_type>& g, thrust::host_vector<real_type>& weights, thrust::host_vector<real_type>& abscissas) const override final
195 {
196 thrust::host_vector<real_type> old = dg::create::abscissas(g);
197 if( m_M == 1)
198 {
199 abscissas = old;
200 weights = dg::evaluate( dg::one, g);
201 return;
202 }
203 abscissas.resize( g.size()*m_M);
204 weights.resize( g.size()*m_M);
205 dg::blas1::copy( 0., weights);
206 unsigned NLeft = m_M/2;
207 double hxleft = (old[0] - g.x0())/(double)(NLeft+1);
208 abscissas[0] = g.x0()+hxleft;
209 weights[0] = hxleft;
210 unsigned idx = 0;
211 std::vector<real_type> sim = simpsons_weights( NLeft+1, hxleft);
212 weights[idx] += sim[0];
213 for( unsigned k=1; k<=NLeft; k++)
214 {
215 idx++;
216 abscissas[idx] = abscissas[idx-1]+hxleft;
217 weights[idx] += sim[k];
218 }
219 // now abs is at old[0]
220 unsigned NMiddle = m_M;
221 for( unsigned i=0; i<old.size()-1; i++)
222 {
223 double hxmiddle = (old[i+1] - old[i])/(double)NMiddle;
224 sim = simpsons_weights( NMiddle+1, hxmiddle);
225 weights[idx] += sim[0];
226 for( unsigned k=1; k<=NMiddle; k++)
227 {
228 idx++;
229 abscissas[idx] = abscissas[idx-1]+hxmiddle;
230 weights[idx] += sim[k];
231 }
232 }
233 unsigned NRight = m_M - 1 - m_M/2;
234 double hxright = (g.x1() - old[g.size()-1] )/(double)(NRight+1);
235 if( NRight > 0)
236 {
237 sim = simpsons_weights( NRight+1, hxright);
238 weights[idx] += sim[0];
239 for( unsigned k=1; k<=NRight; k++)
240 {
241 idx++;
242 abscissas[idx] = abscissas[idx-1]+hxright;
243 weights[idx] += sim[k];
244 }
245 }
246 weights[idx] += hxright;
247 RealGrid1d<real_type> nGrid( g.x0(), g.x1(), g.n(), g.N()*m_M);
248 thrust::host_vector<real_type> wrong_weights = dg::create::weights(nGrid);
249 dg::blas1::pointwiseDivide( wrong_weights, weights, weights);
250 }
251 virtual unsigned do_N_new( unsigned N_old, bc bcx) const override final{
252 return N_old*m_M;
253 }

292 thrust::host_vector<real_type> equidist_ref( unsigned add_x, unsigned node, unsigned n, unsigned N, dg::bc bcx, unsigned howmany) const
293 {
294 assert( howm_ <= N);
295 assert( node_ <= N);
296 if( node_ != 0 && node_ != N)
297 assert( howm_ <= node_ && howm_ <= N-node_);
298 if( add_x_ == 0 || howm_ == 0)
299 {
300 thrust::host_vector<real_type> w_( n*N, 1);
301 return w_;
302 }
303 //there are add_x+1 finer cells per refined cell ...
304 thrust::host_vector< real_type> left( n*N+n*add_x_*howm_, 1), right(left);
305 for( unsigned i=0; i<(add_x_+1)*howm_; i++)//the original cell and the additional ones
306 for( unsigned k=0; k<n; k++)
307 left[i*n+k] = add_x_ + 1;
308 //mirror left into right
309 for( unsigned i=0; i<right.size(); i++)
310 right[i] = left[ (left.size()-1)-i];
311 thrust::host_vector< real_type> both( n*N+2*n*add_x_*howm_, 1);
312 for( unsigned i=0; i<left.size(); i++)
313 both[i] *= left[i];
314 for( unsigned i=0; i<right.size(); i++)
315 both[i+n*add_x_*howm_] *= right[i];
316 if( node_ == 0 && bcx != dg::PER) { return left; }
317 else if( node_ == N && bcx != dg::PER) { return right; }
318 else if((node_ == N || node_ == 0) && bcx == dg::PER) { return both; }
319 else
320 {
321 thrust::host_vector<real_type> w_ = both;
322 //now shift indices so that refinement is around node_s
323 for( unsigned i=0; i<both.size(); i++)
324 w_[((howm_*add_x_+node_)*n+i)%both.size()] = both[i];
325 return w_;
326 }
