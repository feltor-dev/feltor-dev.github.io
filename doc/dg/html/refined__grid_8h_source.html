29 thrust::host_vector<real_type> normalize_weights_and_compute_abscissas( const RealGrid1d<real_type>& g, thrust::host_vector<real_type>& weights)
30 {
31  //normalize weights
32  unsigned Nx_new = weights.size()/g.n();
33  for( unsigned i=0;i<weights.size(); i++)
34  weights[i] *= (real_type)g.N()/(real_type)Nx_new;
35 
36  thrust::host_vector<real_type> boundaries(Nx_new+1), abs(g.n()*Nx_new);
37  boundaries[0] = g.x0();
38  for( unsigned i=0; i<Nx_new; i++)
39  {
40  boundaries[i+1] = boundaries[i] + g.lx()/(real_type)Nx_new/weights[g.n()*i];
41  for( unsigned j=0; j<g.n(); j++)
42  {
43  abs[i*g.n()+j] = (boundaries[i+1]+boundaries[i])/2. +
44  (boundaries[i+1]-boundaries[i])/2.*g.dlt().abscissas()[j];
45  }
46  }
47  return abs;
48 }

175  thrust::host_vector<real_type> equidist_ref( unsigned add_x, unsigned node, unsigned n, unsigned N, dg::bc bcx, unsigned howmany) const
176  {
177  assert( howm_ <= N);
178  assert( node_ <= N);
179  if( node_ != 0 && node_ != N)
180  assert( howm_ <= node_ && howm_ <= N-node_);
181  if( add_x_ == 0 || howm_ == 0)
182  {
183  thrust::host_vector<real_type> w_( n*N, 1);
184  return w_;
185  }
186  //there are add_x+1 finer cells per refined cell ...
187  thrust::host_vector< real_type> left( n*N+n*add_x_*howm_, 1), right(left);
188  for( unsigned i=0; i<(add_x_+1)*howm_; i++)//the original cell and the additional ones
189  for( unsigned k=0; k<n; k++)
190  left[i*n+k] = add_x_ + 1;
191  //mirror left into right
192  for( unsigned i=0; i<right.size(); i++)
193  right[i] = left[ (left.size()-1)-i];
194  thrust::host_vector< real_type> both( n*N+2*n*add_x_*howm_, 1);
195  for( unsigned i=0; i<left.size(); i++)
196  both[i] *= left[i];
197  for( unsigned i=0; i<right.size(); i++)
198  both[i+n*add_x_*howm_] *= right[i];
199  if( node_ == 0 && bcx != dg::PER) { return left; }
200  else if( node_ == N && bcx != dg::PER) { return right; }
201  else if((node_ == N || node_ == 0) && bcx == dg::PER) { return both; }

205  //now shift indices so that refinement is around node_s
