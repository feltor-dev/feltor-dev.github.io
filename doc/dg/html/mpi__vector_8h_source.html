269  int ndims;
270  MPI_Cartdim_get( comm, &ndims);
271  int dims[ndims], periods[ndims], coords[ndims];
272  MPI_Cart_get( comm, ndims, dims, periods, coords);
273  if( dims[direction] == 1) silent_ = true;
274  }
275  //mpi_cart_shift may return MPI_PROC_NULL then the receive buffer is not modified
276  MPI_Cart_shift( comm_, direction_, -1, &m_source[0], &m_dest[0]);
277  MPI_Cart_shift( comm_, direction_, +1, &m_source[1], &m_dest[1]);
278  assert( direction <3);
279  thrust::host_vector<int> hbgather1(buffer_size()), hbgather2(hbgather1), hbscattr1(buffer_size()), hbscattr2(hbscattr1);
280  thrust::host_vector<int> mid_gather( 4*buffer_size()), mid_scatter( 4*buffer_size());

283  case( 0):
284  for( unsigned i=0; i<dim_[2]*dim_[1]; i++)
285  for( unsigned j=0; j<n; j++)

297  mid_scatter[i*4*n+3*n+j] = i*(6*n) + 4*n + j;

299  }

301  case( 1):

303  for( unsigned j=0; j<n; j++)
