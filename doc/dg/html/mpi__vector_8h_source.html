325 void NearestNeighborComm<I,B,V>::construct( unsigned n, const unsigned dimensions[3], MPI_Comm comm, unsigned direction)
326 {
327  static_assert( std::is_base_of<SharedVectorTag, get_tensor_category<V>>::value,
328  "Only Shared vectors allowed");
329  silent_=false;
330  n_=n;
331  dim_[0] = dimensions[0], dim_[1] = dimensions[1], dim_[2] = dimensions[2];
332  direction_ = direction;
333  if( dimensions[2] == 1 && direction == 1) trivial_ = true;
334  else if( direction == 2) trivial_ = true;
335  else trivial_ = false;
336  assert( direction <3);
337  comm_ = comm;
338  //mpi_cart_shift may return MPI_PROC_NULL then the receive buffer is not modified
339  MPI_Cart_shift( comm_, direction_, -1, &m_source[0], &m_dest[0]);
340  MPI_Cart_shift( comm_, direction_, +1, &m_source[1], &m_dest[1]);
341  {
342  int ndims;
343  MPI_Cartdim_get( comm, &ndims);
344  int dims[ndims], periods[ndims], coords[ndims];
345  MPI_Cart_get( comm, ndims, dims, periods, coords);
346  if( dims[direction] == 1) silent_ = true;
347  }
348  if( !silent_)
349  {
350  outer_size_ = dimensions[0]*dimensions[1]*dimensions[2]/buffer_size();
351  assert( outer_size_ > 1 && "Parallelization too fine grained!"); //right now we cannot have that
352  thrust::host_vector<int> mid_gather( 4*buffer_size());
353  switch( direction)
354  {
355  case( 0):
356  for( unsigned i=0; i<dim_[2]*dim_[1]; i++)
357  for( unsigned j=0; j<n; j++)
358  {
359  mid_gather[(0*n+j)*dim_[2]*dim_[1]+i] = i*dim_[0] + j;
360  mid_gather[(1*n+j)*dim_[2]*dim_[1]+i] = i*dim_[0] + n + j;
361  mid_gather[(2*n+j)*dim_[2]*dim_[1]+i] = i*dim_[0] + dim_[0]-2*n + j;
362  mid_gather[(3*n+j)*dim_[2]*dim_[1]+i] = i*dim_[0] + dim_[0]- n + j;
363  }
364  break;
365  case( 1):
366  for( unsigned i=0; i<dim_[2]; i++)
367  for( unsigned j=0; j<n; j++)
368  for( unsigned k=0; k<dim_[0]; k++)
369  {
370  mid_gather[((0*n+j)*dim_[2]+i)*dim_[0] + k] = (i*dim_[1] + j)*dim_[0] + k;
371  mid_gather[((1*n+j)*dim_[2]+i)*dim_[0] + k] = (i*dim_[1] + n + j)*dim_[0] + k;
372  mid_gather[((2*n+j)*dim_[2]+i)*dim_[0] + k] = (i*dim_[1] + dim_[1]-2*n + j)*dim_[0] + k;
373  mid_gather[((3*n+j)*dim_[2]+i)*dim_[0] + k] = (i*dim_[1] + dim_[1]- n + j)*dim_[0] + k;
374  }
375  break;
376  case( 2):
377  for( unsigned i=0; i<n; i++)
378  for( unsigned j=0; j<dim_[0]*dim_[1]; j++)
379  {
380  mid_gather[(0*n+i)*dim_[0]*dim_[1]+j] = (i )*dim_[0]*dim_[1] + j;
381  mid_gather[(1*n+i)*dim_[0]*dim_[1]+j] = (i + n )*dim_[0]*dim_[1] + j;
382  mid_gather[(2*n+i)*dim_[0]*dim_[1]+j] = (i + dim_[2]-2*n )*dim_[0]*dim_[1] + j;
383  mid_gather[(3*n+i)*dim_[0]*dim_[1]+j] = (i + dim_[2]- n )*dim_[0]*dim_[1] + j;
384  }
385  break;
386  }
387  gather_map_middle_ = mid_gather; //transfer to device
388  internal_buffer_.data().resize( 6*buffer_size() );
389  }
