364 void NearestNeighborComm<I,B,V>::construct( unsigned n, const unsigned dimensions[3], MPI_Comm comm, unsigned direction)
365 {
366  static_assert( std::is_base_of<SharedVectorTag, get_tensor_category<V>>::value,
367  "Only Shared vectors allowed");
368  m_silent=false;
369  m_n=n;
370  m_dim[0] = dimensions[0], m_dim[1] = dimensions[1], m_dim[2] = dimensions[2];
371  m_direction = direction;
372  if( dimensions[2] == 1 && direction == 1) m_trivial = true;
373  else if( direction == 2) m_trivial = true;
374  else m_trivial = false;
375  assert( direction <3);
376  m_comm = comm;
377  //mpi_cart_shift may return MPI_PROC_NULL then the receive buffer is not modified
378  MPI_Cart_shift( m_comm, m_direction, -1, &m_source[0], &m_dest[0]);
379  MPI_Cart_shift( m_comm, m_direction, +1, &m_source[1], &m_dest[1]);
380  {
381  int ndims;
382  MPI_Cartdim_get( comm, &ndims);
383  int dims[ndims], periods[ndims], coords[ndims];
384  MPI_Cart_get( comm, ndims, dims, periods, coords);
385  if( dims[direction] == 1) m_silent = true;
386  }
387  if( !m_silent)
388  {
389  m_outer_size = dimensions[0]*dimensions[1]*dimensions[2]/buffer_size();
390  assert( m_outer_size > 1 && "Parallelization too fine grained!"); //right now we cannot have that
391  thrust::host_vector<int> mid_gather( 4*buffer_size());
392  switch( direction)
393  {
394  case( 0):
395  for( unsigned i=0; i<m_dim[2]*m_dim[1]; i++)
396  for( unsigned j=0; j<n; j++)
397  {
398  mid_gather[(0*n+j)*m_dim[2]*m_dim[1]+i] = i*m_dim[0] + j;
399  mid_gather[(1*n+j)*m_dim[2]*m_dim[1]+i] = i*m_dim[0] + n + j;
400  mid_gather[(2*n+j)*m_dim[2]*m_dim[1]+i] = i*m_dim[0] + m_dim[0]-2*n + j;
401  mid_gather[(3*n+j)*m_dim[2]*m_dim[1]+i] = i*m_dim[0] + m_dim[0]- n + j;
402  }
403  break;
404  case( 1):
405  for( unsigned i=0; i<m_dim[2]; i++)
406  for( unsigned j=0; j<n; j++)
407  for( unsigned k=0; k<m_dim[0]; k++)
408  {
409  mid_gather[((0*n+j)*m_dim[2]+i)*m_dim[0] + k] = (i*m_dim[1] + j)*m_dim[0] + k;
410  mid_gather[((1*n+j)*m_dim[2]+i)*m_dim[0] + k] = (i*m_dim[1] + n + j)*m_dim[0] + k;
411  mid_gather[((2*n+j)*m_dim[2]+i)*m_dim[0] + k] = (i*m_dim[1] + m_dim[1]-2*n + j)*m_dim[0] + k;
412  mid_gather[((3*n+j)*m_dim[2]+i)*m_dim[0] + k] = (i*m_dim[1] + m_dim[1]- n + j)*m_dim[0] + k;
413  }
414  break;
415  case( 2):
416  for( unsigned i=0; i<n; i++)
417  for( unsigned j=0; j<m_dim[0]*m_dim[1]; j++)
418  {
419  mid_gather[(0*n+i)*m_dim[0]*m_dim[1]+j] = (i )*m_dim[0]*m_dim[1] + j;
420  mid_gather[(1*n+i)*m_dim[0]*m_dim[1]+j] = (i + n )*m_dim[0]*m_dim[1] + j;
421  mid_gather[(2*n+i)*m_dim[0]*m_dim[1]+j] = (i + m_dim[2]-2*n )*m_dim[0]*m_dim[1] + j;
422  mid_gather[(3*n+i)*m_dim[0]*m_dim[1]+j] = (i + m_dim[2]- n )*m_dim[0]*m_dim[1] + j;
423  }
424  break;
425  }
426  m_gather_map_middle = mid_gather; //transfer to device
427  m_internal_buffer.data().resize( 6*buffer_size() );
428 #ifdef _DG_CUDA_UNAWARE_MPI
429  m_internal_host_buffer.data().resize( 6*buffer_size() );
430 #endif
431  }
