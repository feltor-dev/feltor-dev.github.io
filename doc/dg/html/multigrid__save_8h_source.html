143template<class Nested, class NestedOperator, class ContainerType0, class ContainerType1>
144void nested_iterations(
145 NestedOperator&& op, ContainerType0& x, const ContainerType1& b,
146 std::vector<std::function<void( const ContainerType1&, ContainerType0&)>>
147 inverse_op, Nested& nested
148)
149{
150 // compute residual r = b - A x
151 dg::blas2::symv(op[0], x, nested.r(0));
152 dg::blas1::axpby(1., b, -1.0, nested.r(0));
153 // project residual and x down to coarse grid
154 dg::blas1::copy( x, nested.x(0));
155 unsigned stages = nested.stages();
156 for( unsigned u=0; u<stages-1; u++)
157 {
158 dg::blas2::gemv( nested.projection(u), nested.r(u), nested.r(u+1));
159 dg::blas2::gemv( nested.projection(u), nested.x(u), nested.x(u+1));
160 }
161
162 //now solve residual equations
163 for( unsigned u=stages-1; u>0; u--)
164 {
165 // compute FAS right hand side
166 dg::blas2::symv( op[u], nested.x(u), nested.b(u));
167 dg::blas1::axpby( 1., nested.r(u), +1., nested.b(u));
168 dg::blas1::copy( nested.x(u), nested.r(u)); // save initial x0
169 inverse_op[u]( nested.b(u), nested.x(u));
170 dg::blas1::axpby( 1., nested.x(u), -1., nested.r(u), nested.x(u) );
171 dg::blas2::symv( 1., nested.interpolation(u-1), nested.x(u),
172 1., nested.x(u-1));
173 }
174 //update initial guess
175 dg::blas1::copy( nested.x(0),x);
176 inverse_op[0]( b, x);
177}

180template<class Nested, class NestedOperator,
181 class ContainerType0, class ContainerType1>
182void multigrid_cycle(
183 NestedOperator&& op,
184 std::vector<std::function<void( const ContainerType1&, ContainerType0&)>>
185 inverse_op, Nested& nested,
186 unsigned gamma, unsigned p)
187{
188 // 1 multigrid cycle beginning on grid p
189 // p < stages-1
190 // x[p] initial condition on input, solution on output
191 // x[p+1] write only (solution of lower stage)
192 // b[p] read only,
193 // b[p+1] write only (residual after Pre-smooting at lower stage)
194 // r[p] write only
195 // gamma: typically 1 (V-cycle) or 2 (W-cycle)
196
197 unsigned stages = nested.stages();
198 // compute FAS right hand side
199 dg::blas2::symv( op[p], nested.x(p), nested.b(p));
200 dg::blas1::axpby( 1., nested.r(p), +1., nested.b(p));
201 dg::blas1::copy( nested.x(p), nested.w(p)); // save x0
202 // 1. Pre-Smooth nu times
203 inverse_op[p]( nested.b(p), nested.x(p));
204
205 // 2. Update residuum
206 dg::blas1::axpby( 1., nested.x(p), -1., nested.w(p)); // delta = x - x0
207 dg::blas2::symv( op[p], nested.w(p), nested.b(p));
208 dg::blas1::axpby( 1., nested.b(p), +1., nested.r(p));
209 // 3. Coarsen
210 dg::blas2::symv( nested.projection(p), nested.r(p), nested.r(p+1));
211 dg::blas2::gemv( nested.projection(p), nested.x(p), nested.x(p+1));
212 // 4. Solve or recursive call to get x[p+1] with initial guess 0
213 if( p+1 == stages-1)
214 {
215 // compute FAS right hand side
216 dg::blas2::symv( op[p+1], nested.x(p+1), nested.b(p+1));
217 dg::blas1::axpby( 1., nested.r(p+1), +1., nested.b(p+1));
218 dg::blas1::copy( nested.x(p+1), nested.r(p+1)); // save initial x0
219 inverse_op[p+1]( nested.b(p+1), nested.x(p+1));
220 dg::blas1::axpby( 1., nested.x(p+1), -1., nested.r(p+1), nested.x(p+1) );
221 }
222 else
223 {
224 for( unsigned u=0; u<gamma; u++)
225 multigrid_cycle( std::forward<NestedOperator>(op), inverse_op,
226 nested, gamma, p+1);
227 }
228
229 // 5. Correct and update residuum
230 dg::blas2::symv( nested.interpolation(p), nested.x(p+1), nested.w(p));
231 dg::blas2::symv( op[p], nested.w(p), nested.b(p));
232 dg::blas1::axpby( 1., nested.b(p), +1., nested.r(p));
233 dg::blas1::axpby( 1., nested.w(p), +1., nested.x(p));
234
235 // 6. Post-Smooth nu times
236 inverse_op[p](nested.b(p), nested.x(p));
237}
238template<class Nested, class NestedOperator,
239 class ContainerType0, class ContainerType1>
240void full_multigrid(
241 NestedOperator&& op,
242 std::vector<std::function<void( const ContainerType1&, ContainerType0&)>>
243 inverse_op, Nested& nested,
244 unsigned gamma, unsigned mu)
245{
246 // compute residual r = b - A x
247 dg::blas2::symv(op[0], nested.x(0), nested.r(0));
248 dg::blas1::axpby(1., nested.b(0), -1.0, nested.r(0));
249 // project residual and x down to coarse grid
250 unsigned stages = nested.stages();
251 for( unsigned u=0; u<stages-1; u++)
252 {
253 dg::blas2::gemv( nested.projection(u), nested.r(u), nested.r(u+1));
254 dg::blas2::gemv( nested.projection(u), nested.x(u), nested.x(u+1));
255 }
256 //begins on coarsest level and cycles through to highest
257 unsigned s = nested.stages()-1;
258 // compute FAS right hand side
259 dg::blas2::symv( op[s], nested.x(s), nested.b(s));
260 dg::blas1::axpby( 1., nested.r(s), +1., nested.b(s));
261 dg::blas1::copy( nested.x(s), nested.r(s)); // save initial x0
262 inverse_op[s]( nested.b(s), nested.x(s));
263 dg::blas1::axpby( 1., nested.x(s), -1., nested.r(s), nested.x(s) );
264
265 for( int p=stages-2; p>=0; p--)
266 {
267 dg::blas2::gemv( 1., nested.interpolation(p), nested.x(p+1), 1., nested.x(p));
268 for( unsigned u=0; u<mu; u++)
269 multigrid_cycle( std::forward<NestedOperator>(op), inverse_op,
270 nested, gamma, p);
271 }

737 template<class SymmetricOp, class ContainerType0, class ContainerType1>
738 void pcg_solve( std::vector<SymmetricOp>& op,
739 ContainerType0& x, const ContainerType1& b, std::vector<value_type> ev, unsigned nu_pre, unsigned
740 nu_post, unsigned gamma, value_type eps)
741 {
742
743 dg::blas1::copy(b, m_b[0]);
744 //PCG WITH MULTIGRID CYCLE AS PRECONDITIONER
745 unsigned max_iter_ = m_grids[0]->size();
746 value_type nrmb = sqrt( blas2::dot( op[0].weights(), m_b[0]));
747 if( nrmb == 0)
748 {
749 blas1::copy( m_b[0], x);
750 return;
751 }
752 blas2::symv( op[0],x,m_cgr);
753 blas1::axpby( 1., m_b[0], -1., m_cgr);
754 //if x happens to be the solution
755 if( sqrt( blas2::dot(op[0].weights(),m_cgr) )
756 < eps*(nrmb + 1))
757 return;
758
759 dg::blas1::copy( 0, m_x[0]);
760 dg::blas1::copy( m_cgr, m_b[0]);
761 full_multigrid( op,m_x, m_b, ev, nu_pre, nu_post, gamma, 1, eps);
762 dg::blas1::copy( m_x[0], m_p);
763
764 //and store the scalar product
765 value_type nrmzr_old = blas1::dot( m_p,m_cgr);
766 value_type alpha, nrmzr_new;
767 for( unsigned i=2; i<max_iter_; i++)
768 {
769 blas2::symv( op[0], m_p, m_x[0]);
770 alpha = nrmzr_old/blas1::dot( m_p, m_x[0]);
771 blas1::axpby( alpha, m_p, 1., x);
772 blas1::axpby( -alpha, m_x[0], 1., m_cgr);
773 value_type error = sqrt( blas2::dot(op[0].weights(), m_cgr))/(nrmb+1);
774 //DG_RANK0 std::cout << "\t\t\tError at "<<i<<" is "<<error<<"\n";
775 if( error < eps)
776 return;
777 dg::blas1::copy( 0, m_x[0]);
778 dg::blas1::copy( m_cgr, m_b[0]);
779 full_multigrid( op,m_x, m_b, ev, nu_pre, nu_post, gamma, 1, eps);
780
781 nrmzr_new = blas1::dot( m_x[0], m_cgr);
782 blas1::axpby(1., m_x[0], nrmzr_new/nrmzr_old, m_p );
783 nrmzr_old=nrmzr_new;
784 }
